025 /**
026  * This class just extends the Java2HTML converter and adds a background colour where the lines were executed. Unfortunately most of the interesting
027  * methods were private so there is a lot of copy and paste, pity, se la vie.
028  * 
029  * @author Michael Couck
030  * @since 09.12.09
031  * @version 01.00
032  */
033 public class JavaSource2HTMLConverterExt extends JavaSource2HTMLConverter {
034 
035   // private static Logger LOGGER = Logger.getLogger(JavaSource2HTMLConverterExt.class);
036 
037   public boolean java2HtmlHomepageLinkEnabled = false;
038   private final String HTML_BLOCK_HEADER = "\n\n<!-- ======================================================== -->\n"
039       + "<!-- = Java Sourcecode to HTML automatically converted code = -->\n<!-- =   " + Version.getJava2HtmlConverterTitle() + " "
040       + Version.getBuildDate() + " by Markus Gebhard  markus@jave.de   = -->\n"
041       + "<!-- =     Further information: http://www.java2html.de     = -->\n" + "<div align=\"{0}\" class=\"java\">\n"
042       + "<table border=\"{1}\" cellpadding=\"3\" " + "cellspacing=\"0\" bgcolor=\"{2}\">\n";
043 
044   private Class<?, ?> klass;
045   private String STYLE = "style=\"background-color: #C7D6D7;\"";
046 
047   // private String blue = "#C7D6D7;";
048   // private String yellow = "#FBF4B7";
049   // private String pink = "#FBC6C6";
050   // private String blue = "#C7D6D7";
051 
052   private int lineCifferCount;

054   public JavaSource2HTMLConverterExt(Class<?, ?> klass) {
055     super();
056     this.klass = klass;

059   public void convert(JavaSource source, JavaSourceConversionOptions options, BufferedWriter writer) throws IOException {
060     if (source == null) {
061       throw new IllegalStateException("Trying to write out converted code without having source set.");
062     }
063 
064     String alignValue = getHtmlAlignValue(options.getHorizontalAlignment());
065     String bgcolorValue = options.getStyleTable().get(JavaSourceType.BACKGROUND).getHtmlColor();
066     String borderValue = (options.isShowTableBorder()) ? "2" : "0";
067 
068     writer.write(MessageFormat.format(HTML_BLOCK_HEADER, new Object[] { alignValue, borderValue, bgcolorValue }));
069 
070     if ((options.isShowFileName()) && (source.getFileName() != null)) {
071       writeFileName(source, writer);
072     }
073 
074     writer.write("   <tr>");
075     writer.newLine();
076 
077     writeSourceCode(source, options, writer);
078 
079     writer.write("   </tr>");
080     writer.newLine();
081 
082     if ((options.isShowJava2HtmlLink()) || (java2HtmlHomepageLinkEnabled)) {
083       writer
084           .write("  <!-- start Java2Html link -->\n   <tr>\n    <td align=\"right\">\n<small>\n<a href=\"http://www.java2html.de\" target=\"_blank\">Java2html</a>\n</small>\n    </td>\n   </tr>\n  <!-- end Java2Html link -->\n");
085     }
086     writer
087         .write("</table>\n</div>\n<!-- =       END of automatically generated HTML code       = -->\n<!-- ======================================================== -->\n\n");
088   }
089 
090   private String getHtmlAlignValue(HorizontalAlignment alignment) {
091     final StringHolder stringHolder = new StringHolder();
092     alignment.accept(new IHorizontalAlignmentVisitor() {
093       public void visitLeftAlignment(HorizontalAlignment horizontalAlignment) {
094         stringHolder.setValue("left");
095       }
096 
097       public void visitRightAlignment(HorizontalAlignment horizontalAlignment) {
098         stringHolder.setValue("right");
099       }
100 
101       public void visitCenterAlignment(HorizontalAlignment horizontalAlignment) {
102         stringHolder.setValue("center");
103       }
104     });
105     return stringHolder.getValue();
106   }

108   private void writeFileName(JavaSource source, BufferedWriter writer) throws IOException {
109     writer.write("  <!-- start headline -->\n   <tr>\n    <td colspan=\"2\">\n     <center><font size=\"+2\">\n      <code><b>\n");
110     writer.write(source.getFileName());
111     writer.newLine();
112     writer.write("      </b></code>\n     </font></center>\n    </td>\n   </tr>\n  <!-- end headline -->\n");
113   }
114 
115   private void writeSourceCode(JavaSource source, JavaSourceConversionOptions options, BufferedWriter writer) throws IOException {
116     writer.write("  <!-- start source code -->\n   <td nowrap=\"nowrap\" valign=\"top\" align=\"left\">\n    <code>\n");
117 
118     this.lineCifferCount = String.valueOf(source.getLineCount()).length();
119 
120     JavaSourceIterator iterator = source.getIterator();
121     int lineNumber = 1;
122     while (iterator.hasNext()) {
123       JavaSourceRun run = iterator.getNext();
124       if (run.isAtStartOfLine()) {
125         if (options.isAddLineAnchors()) {
126           writeLineAnchorStart(options, writer, lineNumber);
127         }
128         if (options.isShowLineNumbers()) {
129           writeLineNumber(options, writer, lineNumber);
130         }
131         if (options.isAddLineAnchors()) {
132           writeLineAnchorEnd(writer);
133         }
134         ++lineNumber;
135       }
136 
137       toHTML(options.getStyleTable(), run, writer, lineNumber);
138       if ((run.isAtEndOfLine()) && (iterator.hasNext())) {
139         writer.write("<br />");
140         writer.newLine();
141       }
142     }
143     writer.write("</code>\n    \n   </td>\n  <!-- end source code -->\n");
144   }
145 
146   private void writeLineAnchorEnd(BufferedWriter writer) throws IOException {
147     writer.write("</a>");
148   }
149 
150   private void writeLineAnchorStart(JavaSourceConversionOptions options, BufferedWriter writer, int lineNumber) throws IOException {
151     writer.write("<a name=\"");
152     writer.write(options.getLineAnchorPrefix() + lineNumber);
153     writer.write("\">");
154   }

156   private void writeLineNumber(JavaSourceConversionOptions options, BufferedWriter writer, int lineNo) throws IOException {
157     JavaSourceStyleEntry styleEntry = options.getStyleTable().get(JavaSourceType.LINE_NUMBERS);
158     writeStyleStart(writer, styleEntry, -1);
159 
160     String lineNumber = String.valueOf(lineNo);
161     int cifferCount = this.lineCifferCount - lineNumber.length();
162     while (cifferCount > 0) {
163       writer.write(48);
164       --cifferCount;
165     }
166 
167     writer.write(lineNumber);
168     writeStyleEnd(writer, styleEntry);
169     writer.write("&nbsp;");
170   }
171 
172   private void toHTML(JavaSourceStyleTable styleTable, JavaSourceRun run, BufferedWriter writer, int lineNumber) throws IOException {
173     JavaSourceStyleEntry style = styleTable.get(run.getType());
174 
175     writeStyleStart(writer, style, lineNumber);
176 
177     String t = HtmlUtilities.encode(run.getCode(), "\n ");
178 
179     for (int i = 0; i < t.length(); ++i) {
180       char ch = t.charAt(i);
181       if (ch == ' ') {
182         writer.write("&nbsp;");
183       } else {
184         writer.write(ch);
185       }
186     }
187 
188     writeStyleEnd(writer, style);
189   }
190 
191   private void writeStyleStart(BufferedWriter writer, JavaSourceStyleEntry style, int lineNumber) throws IOException {
192     if (getCovered(this.klass, lineNumber - 1)) {
193       writer.write("<font color=\"" + style.getHtmlColor() + "\" " + STYLE + ">");
194     } else {
195       writer.write("<font color=\"" + style.getHtmlColor() + "\">");
196     }
197     if (style.isBold()) {
198       writer.write("<b>");
199     }
200     if (style.isItalic()) {
201       writer.write("<i>");
202     }
203   }
204 
205   private void writeStyleEnd(BufferedWriter writer, JavaSourceStyleEntry style) throws IOException {
206     if (style.isItalic()) {
207       writer.write("</i>");
208     }
209     if (style.isBold()) {
210       writer.write("</b>");
211     }
212     writer.write("</font>");
213   }

215   private boolean getCovered(Class<?, ?> klass, double lineNumber) {
216     for (Method<?, ?> method : klass.getChildren()) {
217       // LOGGER.warn("Method : " + method);
218       // LOGGER.warn("Children : " + method.getChildren());
219       for (Line<?, ?> line : method.getChildren()) {
220         // LOGGER.warn("Line : " + line + ", line number : " + line.getNumber() + ", number : " + lineNumber);
221         if (line.getNumber() == lineNumber) {
222           boolean lineCovered = line.getCounter() > 0;
223           if (lineCovered) {
224             // LOGGER.warn("Line covered : " + lineCovered);
225           }
226           return lineCovered;
227         }
228       }
229     }
230     for (Class<?, ?> innerKlass : klass.getInnerClasses()) {
231       boolean covered = getCovered(innerKlass, lineNumber);
232       if (covered) {
233         // LOGGER.warn("Covered : " + covered);
234         return true;
235       }
236     }
237     return false;
238   }
