01 package com.ikokoon.serenity.instrumentation.dependency;
02 
03 import org.apache.log4j.Logger;
04 import org.objectweb.asm.AnnotationVisitor;
05 import org.objectweb.asm.Attribute;
06 import org.objectweb.asm.FieldVisitor;
07 import org.objectweb.asm.Type;
08 
09 import com.ikokoon.serenity.Collector;
10 import com.ikokoon.serenity.instrumentation.VisitorFactory;
11 import com.ikokoon.toolkit.Toolkit;
12 
13 /**
14  * Visits and collects the dependency metrics for a field in a class.
15  * 
16  * @author Michael Couck
17  * @since 09.12.09
18  * @version 01.00
19  */
20 public class DependencyFieldAdapter implements FieldVisitor {
21 
22   private Logger logger = Logger.getLogger(this.getClass());
23   /** The parent visitor. */
24   private FieldVisitor visitor;
25   /** The name of the class the field is in. */
26   private String className;
27 
28   /**
29    * Constructor initialises a {@link DependencyFieldAdapter} and takes the parent field visitor, the name of the class the field is in, the
30    * description of the field in byte code and the signature in byte code style.
31    * 
32    * @param visitor
33    *            the parent field visitor
34    * @param className
35    *            the name of the class he field is in
36    * @param description
37    *            the byte code description of the field
38    * @param signature
39    *            the byte code signature of the field
40    */
41   public DependencyFieldAdapter(FieldVisitor visitor, String className, String description, String signature) {
42     this.visitor = visitor;
43     this.className = Toolkit.slashToDot(className);
44     logger.debug("Class name : " + this.className + ", " + description + ", " + signature);
45     VisitorFactory.getSignatureVisitor(this.className, description);
46     if (signature != null) {
47       VisitorFactory.getSignatureVisitor(this.className, signature);
48       String targetClassName = Type.getType(description).getClassName();
49       Collector.collectEfferentAndAfferent(this.className, targetClassName);
50     }
51   }
52 
53   /**
54    * {@inheritDoc}
55    */
56   public AnnotationVisitor visitAnnotation(String desc, boolean visible) {
57     logger.debug("visitAnnotation : " + desc + ", " + visible);
58     AnnotationVisitor annotationVisitor = visitor.visitAnnotation(desc, visible);
59     AnnotationVisitor adapter = VisitorFactory.getAnnotationVisitor(annotationVisitor, className, desc);
60     return adapter;
61   }
62 
63   /**
64    * {@inheritDoc}
65    */
66   public void visitAttribute(Attribute attr) {
67     // We don't care about attributes
68     visitor.visitAttribute(attr);
69   }
70 
71   /**
72    * {@inheritDoc}
73    */
74   public void visitEnd() {
75     // What can we get here?
76     visitor.visitEnd();
77   }
78 
79 }
