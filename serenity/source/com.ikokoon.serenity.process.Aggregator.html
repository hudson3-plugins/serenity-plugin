024 /**
025  * This class aggregates all the totals for the report. For each method there is a total for the number of lines that the method has and a for each
026  * line that is executed there is a line element. So the percentage of lines that were executed is covered lines / total lines * 100. The complexity
027  * total is added up for each method along with the coverage total for each method and divided by the total methods in the class to get the class
028  * average for coverage and complexity.
029  * 
030  * Similarly for the package totals of complexity and coverage the class totals for the package are added up and divided by the number of classes in
031  * the package.
032  * 
033  * Metrics are also gathered for the dependency, afferent and efferent and the abstractness and stability calculated from that.
034  * 
035  * @author Michael Couck
036  * @since 18.07.09
037  * @version 01.00
038  */
039 public class Aggregator extends AProcess implements IConstants {

060     compositeMethods = new HashMap<Composite<?, ?>, Set<Method<?, ?>>>();
061 
062     Project<?, ?> project = (Project<?, ?>) dataBase.find(Project.class, Toolkit.hash(Project.class.getName()));
063     if (project == null) {
064       project = new Project<Object, Object>();
065       dataBase.persist(project);
066     }
067 
068     List<Package> packages = dataBase.find(Package.class);
069     Set<Line<?, ?>> lines = getLines(packages);
070     Set<Method<?, ?>> methods = getMethods(packages);
071 
072     compositeLines.put(project, lines);
073     compositeMethods.put(project, methods);
074   }

087   @SuppressWarnings("unchecked")
088   protected void aggregateProject(Project<?, ?> project) {
089     List<Package> packages = dataBase.find(Package.class);
090     Set<Line<?, ?>> lines = compositeLines.get(project);
091     Set<Method<?, ?>> methods = compositeMethods.get(project);
092 
093     aggregatePackages(packages);
094     project.setTimestamp(new Date());
095 
096     double classes = 0d;
097 
098     double totalComplexity = 0;
099     double executed = 0d;
100     double interfaces = 0;
101     double implementations = 0;
102     Set<Efferent> efference = new TreeSet<Efferent>();
103     Set<Afferent> afference = new TreeSet<Afferent>();
104 
105     for (Method<?, ?> method : methods) {
106       totalComplexity += method.getComplexity();
107     }
108 
109     for (Line<?, ?> line : lines) {
110       if (line.getCounter() > 0) {
111         executed++;
112       }
113     }
114 
115     if (logger.isDebugEnabled()) {
116       logger.debug("Lines : " + lines.size() + ", executed : " + executed);
117     }
118 
119     if (lines.size() > 0) {
120       for (Package<?, ?> pakkage : packages) {
121         interfaces += pakkage.getInterfaces();
122         implementations += pakkage.getImplementations();
123         efference.addAll(pakkage.getEfferent());
124         afference.addAll(pakkage.getAfferent());
125         classes += pakkage.getChildren().size();
126       }
127     }
128 
129     double coverage = getCoverage(lines.size(), executed);// lines.size() > 0 ? (executed / lines.size()) * 100d : 0;
130     double complexity = getComplexity(methods.size(), totalComplexity); // methods.size() > 0 ? totalComplexity / methods.size() : 0;
131 
132     double stability = getStability(efference.size(), afference.size()); // (efferent + afferent) > 0 ? efferent / (efferent + afferent) : 1d;
133     double abstractness = getAbstractness(interfaces, implementations); // (interfaces + implementations) > 0 ? interfaces / (interfaces +
134     // implementations) : 1d;
135     double distance = getDistance(stability, abstractness); // Math.abs(-stability + -abstractness + 1) / Math.sqrt(Math.pow(a, 2) + Math.pow(b,
136     // 2));
137 
138     project.setComplexity(complexity);
139     project.setCoverage(coverage);
140     project.setAbstractness(abstractness);
141     project.setStability(stability);
142     project.setDistance(distance);
143 
144     project.setLines(lines.size());
145     project.setMethods(methods.size());
146     project.setClasses(classes);
147     project.setPackages(packages.size());
148 
149     setPrecision(project);
150   }

152   /**
153    * 1) u = (x3 - x1)(x2 - x1) + (y3 - y1)(y2 - y1) / ||p2 - p1||² <br>
154    * 2) y = mx + c, 0 = ax + by + c, d = |am + bn + c| / sqrt(a² + b²) : d= |-stability + -abstractness + 1| / sqrt(-1² + -1²)
155    * 
156    * @param stability
157    * @param abstractness
158    * @return
159    */
160   private double getDistance(double stability, double abstractness) {
161     double a = -1, b = -1;
162     double distance = Math.abs(-stability + -abstractness + 1) / Math.sqrt(Math.pow(a, 2) + Math.pow(b, 2));
163     return distance;
164   }

166   private double getAbstractness(double interfaces, double implementations) {
167     double abstractness = (interfaces + implementations) > 0 ? interfaces / (interfaces + implementations) : 1d;
168     return abstractness;
169   }
170 
171   private double getStability(double efferent, double afferent) {
172     double stability = (efferent + afferent) > 0 ? efferent / (efferent + afferent) : 1d;
173     return stability;
174   }

176   private double getComplexity(double methods, double totalComplexity) {
177     double complexity = methods > 0 ? totalComplexity / methods : 0;
178     return complexity;
179   }

181   private double getCoverage(double lines, double executed) {
182     double coverage = lines > 0 ? (executed / lines) * 100d : 0;
183     return coverage;
184   }

196   protected void aggregatePackage(Package<?, ?> pakkage) {
197     Set<Line<?, ?>> lines = compositeLines.get(pakkage);
198     Set<Method<?, ?>> methods = compositeMethods.get(pakkage);
199 
200     double interfaces = 0d;
201     double implementations = 0d;
202     double executed = 0d;
203     double methodAccumulatedComplexity = 0d;
204 
205     for (Line<?, ?> line : lines) {
206       if (line.getCounter() > 0) {
207         executed++;
208       }
209     }
210 
211     for (Method<?, ?> method : methods) {
212       methodAccumulatedComplexity += method.getComplexity();
213     }
214 
215     Set<Efferent> efference = new TreeSet<Efferent>();
216     Set<Afferent> afference = new TreeSet<Afferent>();
217 
218     for (Class<?, ?> klass : pakkage.getChildren()) {
219       if (klass.getInterfaze()) {
220         interfaces++;
221       } else {
222         implementations++;
223       }
224       for (Efferent efferent : klass.getEfferent()) {
225         efference.add(efferent);
226       }
227       for (Afferent afferent : klass.getAfferent()) {
228         afference.add(afferent);
229       }
230     }
231 
232     pakkage.setEfferent(efference);
233     pakkage.setAfferent(afference);
234 
235     pakkage.setLines(lines.size());
236     pakkage.setExecuted(executed);
237 
238     double coverage = getCoverage(lines.size(), executed); // lines.size() > 0 ? (linesExecuted / lines.size()) * 100d : 0;
239     double complexity = getComplexity(methods.size(), methodAccumulatedComplexity); // methods.size() > 0 ? methodAccumulatedComplexity /
240     // methods.size() : 1;
241 
242     double abstractness = getAbstractness(interfaces, implementations); // (interfaces + implementations) > 0 ? interfaces / (interfaces +
243     // implementations) : 1;
244 
245     double stability = getStability(efference.size(), afference.size()); // (efferent + afferent) > 0 ? efferent / (efferent + afferent) : 1d;
246 
247     double distance = getDistance(stability, abstractness); // Math.abs(-stability + -abstractness + 1) / Math.sqrt(Math.pow(a, 2) + Math.pow(b,
248     // 2));
249 
250     pakkage.setInterfaces(interfaces);
251     pakkage.setImplementations(implementations);
252     pakkage.setEfference(efference.size());
253     pakkage.setAfference(afference.size());
254 
255     pakkage.setCoverage(coverage);
256     pakkage.setComplexity(complexity);
257     pakkage.setStability(stability);
258     pakkage.setAbstractness(abstractness);
259     pakkage.setDistance(distance);
260   }

271   protected void aggregateClass(Class<?, ?> klass) {

276     double totalComplexity = 0d;

279       if (line.getCounter() > 0d) {
280         executed++;
281       }
282     }
283     for (Method<?, ?> method : methods) {

288       }

290

292     double complexity = getComplexity(methods.size(), totalComplexity); // methods.size() > 0 ? totalComplexity / methods.size() : 1;
293

297     klass.setEfference(klass.getEfferent().size());
298 
299     double stability = getStability(klass.getEfferent().size(), klass.getAfferent().size()); // (efference + afference) > 0 ? efference /

301

354   protected Set<Line<?, ?>> getLines(Class<?, ?> klass, Set<Line<?, ?>> lines) {
355     for (Class<?, ?> innerKlass : klass.getInnerClasses()) {
356       getLines(innerKlass, lines);
357     }
358     for (Method<?, ?> method : klass.getChildren()) {
359       for (Line<?, ?> line : method.getChildren()) {
360         if (!containsLine(lines, line)) {
361           lines.add(line);
362         }
363       }
364     }
365     if (logger.isDebugEnabled()) {
366       logger.info("Class : " + klass + ", line size : " + lines.size());
367     }
368     return lines;

384   private boolean containsLine(Set<Line<?, ?>> lines, Line<?, ?> line) {
385     for (Line<?, ?> setLine : lines) {
386       if (setLine.getNumber() == line.getNumber()) {
387         return true;
388       }
389     }
390     return false;
391   }

393   protected void aggregateMethods(List<Method<?, ?>> methods) {
394     for (Method<?, ?> method : methods) {
395       try {
396         double executed = 0d;
397         // Collect all the lines that were executed
398         for (Line<?, ?> line : method.getChildren()) {
399           if (logger.isDebugEnabled()) {
400             logger.debug("Line covered : " + line + ", " + line.getCounter());
401           }
402           if (line.getCounter() > 0) {
403             executed++;
404           }
405         }
406         if (method.getChildren().size() > 0) {
407           double coverage = getCoverage(method.getChildren().size(), executed); // (executed / method.getChildren().size()) * 100d;
408           method.setCoverage(coverage);
409         }
410       } catch (Exception e) {
411         logger.error("Exception peocessing the method element : " + method.getName(), e);
412       }
413     }
414   }

416   private void setPrecision(Composite<?, ?> composite) {
417     Field[] fields = composite.getClass().getDeclaredFields();
418     for (Field field : fields) {
419       if (double.class.isAssignableFrom(field.getType()) || Double.class.isAssignableFrom(field.getDeclaringClass())) {
420         try {
421           field.setAccessible(true);
422           double value = field.getDouble(composite);
423           value = Toolkit.format(value, PRECISION);
424           field.setDouble(composite, value);
425         } catch (Exception e) {
426           logger.error("Exception accessing the field : " + field, e);
427         }
428       }
429     }
