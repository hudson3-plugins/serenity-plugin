01 package com.ikokoon.serenity.hudson.modeller;
02 
03 import java.lang.reflect.Field;
04 import java.util.ArrayList;
05 import java.util.List;
06 import java.util.Set;
07 import java.util.TreeSet;
08 
09 import com.ikokoon.serenity.model.Composite;
10 import com.ikokoon.serenity.model.IModel;
11 import com.ikokoon.serenity.model.Model;
12 import com.ikokoon.toolkit.Toolkit;
13 
14 /**
15  * This class takes a composite and produces a string base64 representation of the serialised model object for the composite. This modeller is for the
16  * JetChart applet.
17  * 
18  * @author Michael Couck
19  * @since 09.11.09
20  * @version 01.00
21  */
22 public class JetChartModeller implements IModeller {
23 
24   /** To avoid having long numbers we make the precision 2 after the decimal. */
25   private final int PRECISION = 2;
26 
27   /** The string representation in base 64 of the serialised model object. */
28   private String model;
29   private Set<String> fields = new TreeSet<String>();
30   {
31     fields.add("coverage");
32     fields.add("complexity");
33     fields.add("abstractness");
34     fields.add("stability");
35     fields.add("distance");
36   }
37 
38   /**
39    * {@inheritDoc}
40    */
41   public String getModel() {
42     return model;
43   }
44 
45   /**
46    * {@inheritDoc}
47    */
48   public void visit(Class<?> klass, Composite<?, ?>... composites) {
49     String name = null;
50     List<String> legend = new ArrayList<String>();
51     List<ArrayList<Double>> metrics = new ArrayList<ArrayList<Double>>();
52     // First build the legend and the metrics
53     Field[] fields = klass.getDeclaredFields();
54     for (int i = 0; i < fields.length; i++) {
55       Field field = fields[i];
56       if (!this.fields.contains(field.getName())) {
57         continue;
58       }
59       String fieldName = field.getName();
60       String firstLetter = fieldName.substring(0, 1);
61       String firstLetterCapital = firstLetter.toUpperCase();
62       legend.add(fieldName.replaceFirst(firstLetter, firstLetterCapital));
63     }
64     for (Composite<?, ?> composite : composites) {
65       name = (String) Toolkit.getValue(klass, composite, "name");
66       for (int i = 0, index = 0; i < fields.length; i++) {
67         Field field = fields[i];
68         if (!this.fields.contains(field.getName())) {
69           continue;
70         }
71         String fieldName = field.getName();
72         double value = 0;
73         if (composite != null) {
74           value = Toolkit.getValue(Double.class, composite, fieldName);
75         }
76 
77         if (fieldName.equals("abstractness") || fieldName.equals("stability") || fieldName.equals("distance")) {
78           value *= 100;
79         }
80 
81         value = Toolkit.format(value, PRECISION);
82 
83         ArrayList<Double> metric = null;
84         if (metrics.size() > index) {
85           metric = (ArrayList<Double>) metrics.get(index);
86         } else {
87           metric = new ArrayList<Double>();
88           metrics.add(metric);
89         }
90         metric.add(value);
91         index++;
92       }
93     }
94     IModel model = new Model(name, legend, metrics);
95     this.model = Toolkit.serializeToBase64(model);
96   }
97 
98 }
