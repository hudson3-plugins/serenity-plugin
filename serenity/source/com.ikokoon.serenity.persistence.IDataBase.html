004 import java.util.List;
005 import java.util.Map;
006 
007 import org.apache.log4j.Logger;
008

012 /**
013  * This is the persistence interface for the Serenity code coverage and dependency functionality.
014  * 
015  * @author Michael Couck
016  * @since 12.08.09
017  * @version 01.00
018  * 
019  *          -------------------------------------------------------------<br>
020  * @version 01.1 <br>
021  *          Removed the JPA oriented methods in favour of in memory as JPA was hopelessly too slow.
022  *          -------------------------------------------------------------<br>
023  * @version 01.2 <br>
024  *          Added the listeners to the databases so the manager can release them from the map. Changed the IDataBase interface to be generic so
025  *          Neodatis can be integrated easier and modified several methods to include the class that is being selected as most persistence libraries
026  *          will need this for their selection.
027  * @version 01.3 <br>
028  *          Added a find method that can be implemented as a fuzzy search method by the implementation with multiple search criteria. For example
029  *          'where x like n and y in m'.
030  */
031 public interface IDataBase {
032 
033   /**
034    * This class manages the databases if there are more than one. It also listens to events that the databases might throw like a close and so on.
035    * 
036    * @author Michael Couck
037    * @since 12.08.09
038    * @version 01.00
039    */
040   public static class DataBaseManager {
041 
042     private static Logger logger = Logger.getLogger(DataBaseManager.class);
043     /** The map of open databases keyed on the database file name. */
044     private static Map<String, IDataBase> dataBases = new HashMap<String, IDataBase>();
045 
046     /**
047      * Access to all the databases in the current VM.
048      * 
049      * @return all the databases
050      */
051     public static synchronized Map<String, IDataBase> getDataBases() {
052       return dataBases;
053     }
054 
055     /**
056      * Accesses a database. In the case the database is not open one will be instanciated on the database file specified. In the case the database
057      * is open but not the right type of database, the old one will be closed and the new one will be opened on the database file, otherwise the
058      * database is returned.
059      * 
060      * @param <E>
061      *            the database type
062      * @param klass
063      *            the class of database to initialise
064      * @param dataBaseFile
065      *            the database file to open the database on
066      * @param create
067      *            whether to create a new database, i.e. deleting the old database file and creating a new one
068      * @return the database
069      */
070     public static synchronized final <E extends IDataBase> IDataBase getDataBase(Class<E> klass, String dataBaseFile, boolean create,
071         IDataBase internalDataBase) {
072       IDataBase dataBase = dataBases.get(dataBaseFile);
073       IDataBaseListener dataBaseListener = getDataBaseListener();
074 
075       if (dataBase == null || dataBase.isClosed()) {
076         dataBase = ObjectFactory.getObject(klass, internalDataBase, dataBaseListener, dataBaseFile, create);
077         logger.debug("Adding database : " + dataBase);
078         dataBases.put(dataBaseFile, dataBase);
079       } else {
080         if (!klass.isAssignableFrom(dataBase.getClass())) {
081           dataBase.close();
082           dataBase = ObjectFactory.getObject(klass, internalDataBase, dataBaseListener, dataBaseFile, create);
083           logger.debug("Adding database : " + dataBase);
084           dataBases.put(dataBaseFile, dataBase);
085         }
086       }
087       return dataBase;
088     }
089 
090     private static IDataBaseListener getDataBaseListener() {
091       IDataBaseListener dataBaseListener = new IDataBaseListener() {
092         public void fireDataBaseEvent(IDataBaseEvent dataBaseEvent) {
093           if (dataBaseEvent.getEventType().equals(IDataBaseEvent.Type.DATABASE_CLOSE)) {
094             IDataBase dataBase = dataBaseEvent.getDataBase();
095             if (!dataBases.values().remove(dataBase)) {
096               logger.warn("Database not removed : " + dataBase);
097             } else {
098               logger.info("Removed database : " + dataBase);
099             }
100           }
101         }
102       };
103       return dataBaseListener;
104     }
105 
106   }
107 
108   /**
109    * Persists an object to a persistent store.
110    * 
111    * @param composite
112    *            the composite to persist
113    * @return the composite that is persisted, typically the id will be set by the underlying implementation
114    */
115   public <E extends Composite<?, ?>> E persist(E composite);
116 
117   /**
118    * Selects a composite based on the class type and the id of the class.
119    * 
120    * @param <E>
121    *            the return type of the class
122    * @param klass
123    *            the type of class to select
124    * @param id
125    *            the unique id of the class
126    * @return the composite with the specified id
127    */
128   public <E extends Composite<?, ?>> E find(Class<E> klass, Long id);
129 
130   /**
131    * Selects a class based on the combination of field values in the parameter list.
132    * 
133    * @param <E>
134    *            the return type of the class
135    * @param klass
136    *            the type of class to select
137    * @param parameters
138    *            the unique combination of field values to select the class with
139    * @return the composite with the specified unique field combination
140    */
141   public <E extends Composite<?, ?>> E find(Class<E> klass, List<Object> parameters);
142 
143   /**
144    * Selects a list of objects based on the values in the objects and the class of the object. The implementations can implement multiple search
145    * criteria including fuzzy selection etc.
146    * 
147    * @param <E>
148    *            the return type of the class
149    * @param klass
150    *            the type of class to select
151    * @param parameters
152    *            the parameters to do the selection with. These are the fields in the objects and the values
153    * @return the list of composites that match the selection criteria
154    */
155   public <E extends Composite<?, ?>> List<E> find(Class<E> klass, Map<String, Object> parameters);
156 
157   /**
158    * Selects all the classes of a particular type. Note this could potentially return the whole database.
159    * 
160    * @param <E>
161    *            the return type of the class
162    * @param klass
163    *            the type of class to select
164    * @return a list of all the objects in the database that have the specified class type
165    */
166   public <E extends Composite<?, ?>> List<E> find(Class<E> klass);
167 
168   /**
169    * Removes an object from the database and returns the removed object as a convenience.
170    * 
171    * @param <E>
172    *            the return type of the class
173    * @param klass
174    *            the type of class to select
175    * @param id
176    *            the unique id of the class
177    * @return the composite with the specified id
178    */
179   public <E extends Composite<?, ?>> E remove(Class<E> klass, Long id);
180 
181   /**
182    * Checks the open status of the database.
183    * 
184    * @return whether the database is open or closed
185    */
186   public boolean isClosed();
187 
188   /**
189    * Closes the database.
190    */
191   public void close();
