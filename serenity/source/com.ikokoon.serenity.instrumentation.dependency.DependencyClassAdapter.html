005

007 import org.objectweb.asm.AnnotationVisitor;
008 import org.objectweb.asm.Attribute;

010 import org.objectweb.asm.ClassVisitor;
011 import org.objectweb.asm.FieldVisitor;

019 /**
020  * This is the entry point for parsing the byte code and collecting the dependency metrics for the class. This class also collects the Java source for
021  * the class if it is available.
022  * 
023  * Dependency metrics consist of the following:<br>
024  * 
025  * 1) Afferent - the number of packages that rely on this package, i.e. how many times it is referenced by other packages, the number of packages that
026  * this class affects<br>
027  * 2) Efferent - the number of packages this package relies on, i.e. the opposite of afferent, the number of classes that this class is effected by<br>
028  * 3) Abstractness - the ratio of abstract to implementations in a package<br>
029  * 4) Entropy - package A relies on package B. Then Package C is introduced and relies on A and B increasing the entropy<br>
030  * 5) Stability - Ce / (Ca + Ce), efferent coupling divided by the afferent coupling plus the efferent coupling<br>
031  * 6) Distance from main - find the stability distance of the package from the main which is (X=0,Y=1) to (X=1,Y=0) <br>
032  * 
033  * @author Michael Couck
034  * @since 18.07.09
035  * @version 01.00
036  */
037 public class DependencyClassAdapter extends ClassAdapter implements Opcodes {
038 
039   /** The logger for the class. */
040   private Logger logger = Logger.getLogger(DependencyClassAdapter.class);
041   /** The name of the class to collect dependency metrics on. */
042   private String className;
043   /** The source for the class if available. */
044   private ByteArrayOutputStream source;
045 
046   /**
047    * Constructor initialises a {@link DependencyClassAdapter} and takes the parent visitor and the name of the class that will be analysed for
048    * dependency.
049    * 
050    * @param classVisitor
051    *            the parent visitor for the class
052    * @param className
053    *            the name of the class to be analysed
054    * @param source
055    *            the Java source
056    */
057   public DependencyClassAdapter(ClassVisitor classVisitor, String className, ByteArrayOutputStream source) {
058     super(classVisitor);
059     this.className = Toolkit.slashToDot(className);
060     this.source = source;
061     logger.debug("Class name : " + className + ", source : " + source);
062   }
063 
064   /**
065    * {@inheritDoc}
066    */
067   public void visit(int version, int access, String className, String signature, String superName, String[] interfaces) {
068     if (logger.isDebugEnabled()) {
069       logger.debug("visit : " + version + ", " + access + ", " + className + ", " + signature + ", " + superName);
070       if (interfaces != null) {
071         logger.debug(Arrays.asList(interfaces).toString());
072       }
073     }
074     String[] normedInterfaces = new String[interfaces.length];
075     for (int i = 0; i < interfaces.length; i++) {
076       String normedInterface = Toolkit.slashToDot(interfaces[i]);
077       normedInterfaces[i] = normedInterface;
078     }
079     Collector.collectEfferentAndAfferent(Toolkit.slashToDot(className), Toolkit.slashToDot(superName));
080     Collector.collectEfferentAndAfferent(Toolkit.slashToDot(className), normedInterfaces);
081     Collector.collectInterface(Toolkit.slashToDot(className), access);
082     super.visit(version, access, className, signature, superName, interfaces);
083   }
084 
085   /**
086    * {@inheritDoc}
087    */
088   public AnnotationVisitor visitAnnotation(String desc, boolean visible) {
089     if (logger.isDebugEnabled()) {
090       logger.debug("visitAnnotation : " + desc + ", " + visible);
091     }
092     AnnotationVisitor visitor = super.visitAnnotation(desc, visible);
093     AnnotationVisitor adapter = VisitorFactory.getAnnotationVisitor(visitor, className, desc);
094     return adapter;
095   }
096 
097   /**
098    * {@inheritDoc}
099    */
100   public void visitAttribute(Attribute attr) {
101     if (logger.isDebugEnabled()) {
102       logger.debug("visitAttribute : " + attr);
103     }
104     // Attributes are code added that is not standard, and we are not really interested in it are we?
105     super.visitAttribute(attr);
106   }
107 
108   /**
109    * {@inheritDoc}
110    */
111   public FieldVisitor visitField(int access, String fieldName, String desc, String signature, Object value) {
112     if (logger.isDebugEnabled()) {
113       logger.debug("visitField : " + access + ", " + fieldName + ", " + desc + ", " + signature + ", " + value);
114     }
115     FieldVisitor visitor = super.visitField(access, fieldName, desc, signature, value);
116     FieldVisitor adapter = VisitorFactory.getFieldVisitor(visitor, DependencyFieldAdapter.class, className, desc, signature);
117     return adapter;
118   }
119 
120   /**
121    * {@inheritDoc}
122    */
123   public void visitInnerClass(String innerName, String outerName, String innerSimpleName, int access) {
124     if (logger.isDebugEnabled()) {
125       logger.info("visitInnerClass : inner name : " + innerName + ", outer name : " + outerName + ", inner simple name : " + innerSimpleName);
126     }
127     if (outerName != null) {
128       Collector.collectInnerClass(Toolkit.slashToDot(innerName), Toolkit.slashToDot(outerName));
129     }
130     super.visitInnerClass(innerName, outerName, innerSimpleName, access);
131   }
132 
133   /**
134    * {@inheritDoc}
135    */
136   public void visitOuterClass(String outerName, String outerMethodName, String outerMethodDescription) {
137     if (logger.isDebugEnabled()) {
138       logger.info("visitOuterClass : class name : " + className + ", owner : " + outerName + ", method name : " + outerMethodName
139           + ", description : " + outerMethodDescription);
140     }
141     Collector.collectOuterClass(className, Toolkit.slashToDot(outerName), outerMethodName, outerMethodDescription);
142     super.visitOuterClass(outerName, outerMethodName, outerMethodDescription);
143   }
144 
145   /**
146    * {@inheritDoc}
147    */
148   public MethodVisitor visitMethod(int access, String name, String desc, String signature, String[] exceptions) {
149     if (logger.isDebugEnabled()) {
150       logger.debug("visitMethod : " + access + ", " + name + ", " + desc + ", " + signature);
151       if (exceptions != null) {
152         logger.debug(Arrays.asList(exceptions).toString());
153       }
154     }
155     MethodVisitor visitor = super.visitMethod(access, name, desc, signature, exceptions);
156     if (exceptions != null) {
157       for (String exception : exceptions) {
158         Collector.collectEfferentAndAfferent(className, Toolkit.slashToDot(exception));
159       }
160     }
161     MethodVisitor adapter = VisitorFactory.getMethodVisitor(visitor, DependencyMethodAdapter.class, className, name, desc);
162     return adapter;
163   }
164 
165   /**
166    * {@inheritDoc}
167    */
168   public void visitSource(String source, String debug) {
169     if (logger.isDebugEnabled()) {
170       logger.debug("visitSource : " + source + ", " + debug);
171     }
172     if (this.source != null && this.source.size() > 0) {
173       Collector.collectSource(className, this.source.toString());
174     }
175     super.visitSource(source, debug);
176   }
177 
178   /**
179    * {@inheritDoc}
180    */
181   public void visitEnd() {
182     logger.debug("visitEnd : ");
183     super.visitEnd();
184   }
