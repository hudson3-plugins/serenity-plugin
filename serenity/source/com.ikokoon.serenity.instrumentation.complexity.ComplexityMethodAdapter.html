001 package com.ikokoon.serenity.instrumentation.complexity;
002 
003 import org.apache.log4j.Logger;
004 import org.objectweb.asm.Label;
005 import org.objectweb.asm.MethodAdapter;
006 import org.objectweb.asm.MethodVisitor;
007 
008 import com.ikokoon.serenity.Collector;
009 import com.ikokoon.serenity.instrumentation.coverage.CoverageMethodAdapter;
010 import com.ikokoon.toolkit.Toolkit;
011 
012 /**
013  * TODO - add the interesting methods to the collection of the complexity. Do we need to add try catch? And what about multiple catch? One for each
014  * potential exception thrown? No?
015  * 
016  * This class just visits the byte code in the classes and collects the complexity metrics for the class. Complexity is calculated by adding one every
017  * time there is a jump instruction.
018  * 
019  * @author Michael Couck
020  * @since 12.07.09
021  * @version 01.00
022  */
023 public class ComplexityMethodAdapter extends MethodAdapter {
024 
025   /** The logger for the class. */
026   private Logger logger = Logger.getLogger(CoverageMethodAdapter.class);
027 
028   /** The name of the class that this method adapter is enhancing the methods for. */
029   private String className;
030   /** The name of the method that is being enhanced. */
031   private String methodName;
032   /** The description of the method being enhanced. */
033   private String methodDescription;
034 
035   /**
036    * The complexity counter, start with one and increment for each jump instruction/decision point. This will give the approximate value of the
037    * McCabe method:<br>
038    * 
039    * M = E âàí N + 2P where
040    * 
041    * M = cyclomatic complexity <br>
042    * E = the number of edges of the graph<br>
043    * N = the number of nodes of the graph<br>
044    * P = the number of connected components.<br>
045    */
046   private int complexityCounter = 1;
047 
048   /**
049    * The constructor initialises a {@link ComplexityMethodAdapter} that takes all the interesting items for the method that is to be enhanced
050    * including the parent method visitor.
051    * 
052    * @param methodVisitor
053    *            the method visitor of the parent
054    * @param className
055    *            the name of the class the method belongs to
056    * @param methodName
057    *            the name of the method that will be collected for complexity
058    * @param methodDescription
059    *            the description of the method, i.e. the byte code signature
060    */
061   public ComplexityMethodAdapter(MethodVisitor methodVisitor, String className, String methodName, String methodDescription) {
062     super(methodVisitor);
063     this.className = Toolkit.slashToDot(className);
064     this.methodName = methodName;
065     this.methodDescription = methodDescription;
066     if (logger.isDebugEnabled()) {
067       logger.debug("Class name : " + className + ", name : " + methodName + ", desc : " + methodDescription);
068     }
069   }
070 
071   /**
072    * {@inheritDoc}
073    */
074   public void visitLineNumber(int lineNumber, Label label) {
075     if (logger.isDebugEnabled()) {
076       logger.debug("visitLineNumber : " + lineNumber + ", " + label + ", " + label.getOffset() + ", " + className + ", " + methodName);
077     }
078     this.mv.visitLineNumber(lineNumber, label);
079   }
080 
081   /**
082    * {@inheritDoc}
083    */
084   public void visitJumpInsn(int opcode, Label paramLabel) {
085     if (logger.isDebugEnabled()) {
086       logger.debug("visitJumpInsn:" + opcode);
087     }
088     complexityCounter++;
089     this.mv.visitJumpInsn(opcode, paramLabel);
090   }
091 
092   /**
093    * {@inheritDoc}
094    */
095   public void visitEnd() {
096     if (logger.isDebugEnabled()) {
097       logger.debug("visitEnd:" + className + ", " + methodName + ", " + methodDescription/* + ", " + lineCounter */);
098     }
099     Collector.collectComplexity(className, methodName, methodDescription, complexityCounter/* , lineCounter */);
100     this.mv.visitEnd();
101   }
102 
103   /**
104    * {@inheritDoc}
105    */
106   public void visitTryCatchBlock(Label start, Label end, Label handler, String type) {
107     logger.debug("visitTryCatchBlock : " + className + ", " + methodName);
108     complexityCounter++;
109     this.mv.visitTryCatchBlock(start, end, handler, type);
110   }
111 
112   /**
113    * {@inheritDoc}
114    */
115   public void visitLookupSwitchInsn(Label dflt, int keys[], Label labels[]) {
116     logger.debug("visitlookupSwitchInst : " + className + ", " + methodName);
117     complexityCounter++;
118     this.mv.visitLookupSwitchInsn(dflt, keys, labels);
119   }
120 
121   /**
122    * {@inheritDoc}
123    */
124   public void visitTableSwitchInsn(int min, int max, Label dflt, Label labels[]) {
125     logger.debug("visitTableSwitchInst : " + className + ", " + methodName);
126     complexityCounter++;
127     this.mv.visitTableSwitchInsn(min, max, dflt, labels);
128   }
129 
130   /**
131    * {@inheritDoc}
132    */
133   public void visitInsn(int opcode) {
134     logger.debug("visitInst : " + className + ", " + methodName);
135     // I could be an ATHROW. Do we count as a jump instruction?
136     this.mv.visitInsn(opcode);
137   }
138 
139 }
