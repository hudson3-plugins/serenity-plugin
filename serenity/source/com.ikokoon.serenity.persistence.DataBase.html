06 /**
07  * This is the top level class in the heirachy for the database classes. Any common methods that are used in the database classes can be added to this
08  * class.
09  * 
10  * @author Michael Couck
11  * @since 01.12.09
12  * @version 01.00
13  */
14 public abstract class DataBase implements IDataBase {
15 
16   /**
17    * Sets the id for a composite.
18    * 
19    * @param composite
20    *            the composite to set the id for
21    */
22   synchronized final void setId(Composite<?, ?> composite) {
23     if (composite == null) {
24       return;
25     }
26     if (composite.getId() == null) {
27       Object[] uniqueValues = Toolkit.getUniqueValues(composite);
28       Long id = Toolkit.hash(uniqueValues);
29       composite.setId(id);
30     }
31   }
32 }
