001 package com.ikokoon.serenity.hudson;
002 
003 import hudson.model.AbstractProject;
004 import hudson.model.Actionable;
005 import hudson.model.ProminentProjectAction;
006 import hudson.model.Run;
007 
008 import java.io.IOException;
009 
010 import org.apache.log4j.Logger;
011 import org.kohsuke.stapler.StaplerRequest;
012 import org.kohsuke.stapler.StaplerResponse;
013 
014 /**
015  * An object in the chain of proxy objects that serve the front end in Hudson.
016  * 
017  * @author Michael Couck
018  * @since 09.12.09
019  * @version 01.00
020  */
021 @SuppressWarnings("unchecked")
022 public class SerenityProjectAction extends Actionable implements ProminentProjectAction {
023 
024   private Logger logger = Logger.getLogger(SerenityProjectAction.class);
025   /** The real owner that generated the build. */
026   private AbstractProject owner;
027 
028   /**
029    * Constructor takes the real build from Hudson.
030    * 
031    * @param owner
032    *            the build that generated the actual build
033    */
034   public SerenityProjectAction(AbstractProject owner) {
035     logger.debug("SerenityProjectAction:");
036     this.owner = owner;
037   }
038 
039   /**
040    * {@inheritDoc}
041    */
042   public String getDisplayName() {
043     logger.debug("getDisplayName");
044     return "Serenity report";
045   }
046 
047   /**
048    * {@inheritDoc}
049    */
050   public String getIconFileName() {
051     logger.debug("getIconFileName");
052     return "graph.gif";
053   }
054 
055   /**
056    * {@inheritDoc}
057    */
058   public String getUrlName() {
059     logger.debug("getUrlName");
060     return "serenity";
061   }
062 
063   /**
064    * {@inheritDoc}
065    */
066   public String getSearchUrl() {
067     logger.debug("getSearchUrl");
068     return getUrlName();
069   }
070 
071   public ISerenityResult getLastResult() {
072     logger.warn("getLastResult");
073     Run build = owner.getLastStableBuild();
074     if (build == null) {
075       build = owner.getLastBuild();
076     }
077     if (build != null) {
078       SerenityBuildAction action = build.getAction(SerenityBuildAction.class);
079       if (action != null) {
080         return action.getResult();
081       }
082     }
083     return null;
084   }
085 
086   public void doIndex(StaplerRequest req, StaplerResponse rsp) throws IOException {
087     logger.debug("doIndex");
088     if (hasResult()) {
089       rsp.sendRedirect2("../lastBuild/serenity");
090     } else {
091       rsp.sendRedirect2("nocoverage");
092     }
093   }
094 
095   public boolean hasResult() {
096     logger.debug("hasResult");
097     return getLastResult() != null;
098   }
099 
100 }
