014 /**
015  * This is the in memory database. Several options were explored including DB4O, Neodatis, JPA, SQL, and finally none were performant enough. As well
016  * as that several hybrids were investigated like including a l1 cache and a l2 cache, but the actual persistence in the case of JPA and SQL was just
017  * too slow. This class does everything in memory and commits the data finally to the under lying database once the processing is finished.
018  * 
019  * @author Michael Couck
020  * @since 11.10.09
021  * @version 01.00
022  */
023 public final class DataBaseRam extends DataBase {
024 
025   private Logger logger = Logger.getLogger(this.getClass());
026   /** The underlying persistence database to the file system. */
027   private IDataBase dataBase;
028   /** The listener that catches events. */
029   private IDataBaseListener dataBaseListener;
030   /** The closed flag. */
031   private boolean closed = true;
032 
033   /** This is the index of packages, classes, methods and lines. */
034   private transient volatile List<Composite<?, ?>> index = new ArrayList<Composite<?, ?>>();
035 
036   /**
037    * Constructor takes the underlying database that will commit the data to the file system, the listener for when the database closes we can
038    * release the resources and whether to create a new database or open an old one.
039    * 
040    * @param dataBase
041    * @param dataBaseListener
042    */
043   DataBaseRam(IDataBase dataBase, IDataBaseListener dataBaseListener) {
044     logger.info("Opening RAM database with " + dataBase + " underneath.");
045     this.dataBase = dataBase;
046     this.dataBaseListener = dataBaseListener;
047     index.clear();
048     closed = false;
049   }

076   /**
077    * {@inheritDoc}
078    */
079   @SuppressWarnings("unchecked")
080   public synchronized final <E extends Composite<?, ?>> List<E> find(Class<E> klass) {
081     List<E> list = new ArrayList<E>();
082     for (Composite<?, ?> composite : index) {
083       if (klass.isInstance(composite)) {
084         list.add((E) composite);
085       }
086     }
087     return list;
088   }

090   /**
091    * {@inheritDoc}
092    */
093   @SuppressWarnings("unchecked")
094   public synchronized final <E extends Composite<?, ?>> E remove(Class<E> klass, Long id) {
095     Composite<?, ?> composite = find(klass, id);
096     if (composite != null) {
097       Composite<?, ?> parent = composite.getParent();
098       if (parent != null) {
099         List<?> children = parent.getChildren();
100         if (children != null) {
101           children.remove(composite);
102         }
103       }
104       composite.setParent(null);
105       if (!index.remove(composite)) {
106         logger.warn("Didn't remove composite with id : " + id + ", because it wasn't in the index.");
107       }
108     }
109     return (E) composite;
110   }

119   /**
120    * {@inheritDoc}
121    */
122   public synchronized final void close() {
123     if (closed) {
124       logger.info("User tried to close the database again");
125       return;
126     }
127     logger.info("Comitting and closing the database");
128 
129     try {
130       logger.info("Persisting index : " + dataBase);
131       if (dataBase != null) {
132         for (Composite<?, ?> composite : index) {
133           if (Package.class.isInstance(composite) || Project.class.isInstance(composite)) {
134             logger.debug("Persisting : " + composite);
135             dataBase.persist(composite);
136           }
137         }
138         dataBase.close();
139       } else {
140         logger.warn("Persistence database was null : " + this);
141       }
142       index.clear();
143       final IDataBase dataBase = this;
144       IDataBaseEvent dataBaseEvent = new IDataBaseEvent() {
145         public IDataBase getDataBase() {
146           return dataBase;
147         }
148 
149         public Type getEventType() {
150           return IDataBaseEvent.Type.DATABASE_CLOSE;
151         }
152       };
153       dataBaseListener.fireDataBaseEvent(dataBaseEvent);
154     } catch (Exception e) {
155       logger.error("Exception comitting and closing the database", e);
156     }
157     closed = true;
158   }

160   /**
161    * This method sets the ids in a graph of objects. The objects need to be stored, perhaps using the top level object in the heirachy, then the
162    * database is consulted for it's uid for the object. The uid is set in the field that has the Identifier annotation on the setter method for the
163    * field.
164    * 
165    * @param <T>
166    *            the type of object
167    * @param object
168    *            the object to set the id for
169    * @param list
170    *            a list of already set id fields
171    */
172   @SuppressWarnings("unchecked")
173   synchronized final <T> void setIds(Composite<?, ?> composite) {
174     if (composite == null) {
175       return;
176     }
177     super.setId(composite);
178     // Insert the object into the index
179     insert(index, composite);
180     logger.debug("Persisted object : " + composite);
181     if (composite instanceof com.ikokoon.serenity.model.Class) {
182       String name = ((com.ikokoon.serenity.model.Class) composite).getName();
183       if (name.indexOf('/') > -1) {
184         Thread.dumpStack();
185       }
186     }
187     List<Composite<?, ?>> children = (List<Composite<?, ?>>) composite.getChildren();
188     for (Composite<?, ?> child : children) {
189       setIds(child);
190     }
191   }

193   /**
194    * Binary search through the index of composite objects.
195    * 
196    * @param index
197    * @param id
198    * @return
199    */
200   @SuppressWarnings("unchecked")
201   final <E extends Composite<?, ?>> E search(List<Composite<?, ?>> index, long id) {
202     int low = 0;
203     int high = index.size() - 1;
204     while (low <= high) {
205       int mid = (low + high) >>> 1;
206       Composite<?, ?> composite = index.get(mid);
207       long midVal = composite.getId();
208       if (midVal < id) {
209         low = mid + 1;
210       } else if (midVal > id) {
211         high = mid - 1;
212       } else {
213         return (E) composite;
214       }
215     }
216     return null;
217   }

219   /**
220    * Insert the composite into the index at the correct index.
221    * 
222    * @param index
223    * @param toInsert
224    */
225   final void insert(List<Composite<?, ?>> index, Composite<?, ?> toInsert) {
226     boolean inserted = false;
227     if (index.size() == 0) {
228       index.add(toInsert);
229       inserted = true;
230     } else {
231       long key = toInsert.getId();
232       int low = 0;
233       int high = index.size();
234       while (low <= high) {
235         int mid = (low + high) >>> 1;
236         if (mid >= index.size()) {
237           index.add(mid, toInsert);
238           inserted = true;
239           break;
240         }
241         Composite<?, ?> composite = index.get(mid);
242         long midVal = composite.getId();
243         if (midVal < key) {
244           int next = mid + 1;
245           if (index.size() > next) {
246             Composite<?, ?> nextComposite = index.get(next);
247             long nextVal = nextComposite.getId();
248             if (nextVal > key) {
249               index.add(next, toInsert);
250               inserted = true;
251               break;
252             }
253           }
254           low = mid + 1;
255         } else if (midVal > key) {
256           int previous = mid - 1;
257           if (previous >= 0) {
258             Composite<?, ?> previousComposite = index.get(previous);
259             long previousVal = previousComposite.getId();
260             if (previousVal < key) {
261               index.add(mid, toInsert);
262               inserted = true;
263               break;
264             }
265           } else {
266             index.add(0, toInsert);
267             inserted = true;
268             break;
269           }
270           high = mid - 1;
271         } else {
272           break;
273         }
274       }
275     }
276     logger.debug("Inserted : " + toInsert + " - " + inserted);
277   }
