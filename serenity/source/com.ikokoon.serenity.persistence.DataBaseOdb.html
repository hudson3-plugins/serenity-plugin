022 /**
023  * This is the database class using Neodatis as the persistence tool.
024  * 
025  * @author Michael Couck
026  * @since 01.12.09
027  * @version 01.00
028  */
029 public class DataBaseOdb extends DataBase {
030 
031   private Logger logger = Logger.getLogger(this.getClass());
032   /** The Neodatis object database for persistence. */
033   private ODB odb = null;
034   /** The listener that catches events. */
035   private IDataBaseListener dataBaseListener;
036   /** The database file for Neodatis. */
037   private String dataBaseFile;
038   /** The closed flag. */
039   private boolean closed = true;
040 
041   /**
042    * Constructor initialises a {@link DataBaseOdb} object.
043    * 
044    * @param dataBaseFile
045    *            the file to open the database with
046    * @param create
047    *            whether to create a new database essentially deleting the database file and creating a new one or to use the data in the existing
048    *            database file
049    */
050   DataBaseOdb(IDataBaseListener dataBaseListener, String dataBaseFile, Boolean create) {
051     this.dataBaseListener = dataBaseListener;
052     this.dataBaseFile = dataBaseFile;
053     logger.info("Opening ODB database on file : " + dataBaseFile);
054     try {
055       if (create) {
056         File file = new File(this.dataBaseFile);
057         if (!file.delete()) {
058           logger.warn("Couldn't delete old database file");
059         }
060         logger.info("Database file : " + file.getAbsolutePath());
061       }
062       odb = ODBFactory.open(this.dataBaseFile);
063     } catch (Exception e) {
064       logger.error("Exception initialising the database", e);
065     }
066     closed = false;
067     commit();
068   }
069 
070   /**
071    * {@inheritDoc}
072    */
073   @SuppressWarnings("unchecked")
074   public synchronized <E extends Composite<?, ?>> E find(Class<E> klass, Long id) {
075     IQuery query = new CriteriaQuery(klass, Where.equal("id", id));
076     return (E) find(query);
077   }
078 
079   /**
080    * {@inheritDoc}
081    */
082   public synchronized <E extends Composite<?, ?>> E find(Class<E> klass, List<Object> parameters) {
083     Long id = Toolkit.hash(parameters.toArray());
084     return find(klass, id);
085   }
086 
087   /**
088    * {@inheritDoc}
089    */
090   @SuppressWarnings("unchecked")
091   public synchronized <E extends Composite<?, ?>> List<E> find(Class<E> klass) {
092     List<E> list = new ArrayList<E>();
093     try {
094       Objects objects = odb.getObjects(klass);
095       while (objects.hasNext()) {
096         list.add((E) objects.next());
097       }
098     } catch (Exception e) {
099       logger.error("Exception selecting objects with class : " + klass, e);
100     }
101     return list;
102   }
103 
104   /**
105    * {@inheritDoc}
106    */
107   @SuppressWarnings("unchecked")
108   public <E extends Composite<?, ?>> List<E> find(final Class<E> klass, final Map<String, Object> parameters) {
109     Set<E> set = new TreeSet<E>();
110     for (String field : parameters.keySet()) {
111       Object value = parameters.get(field);
112       logger.warn("Field : " + field + ", " + value);
113       IQuery query = new CriteriaQuery(klass, Where.like(field, "%" + value.toString() + "%"));
114       try {
115         Objects objects = odb.getObjects(query);
116         logger.warn("Objects : " + objects);
117         if (set.size() == 0) {
118           set.addAll(objects);
119         }
120         set.retainAll(objects);
121         logger.warn("Set : " + set);
122       } catch (Exception e) {
123         logger.error("Exception selecting objects with class : " + klass + ", parameters : " + parameters, e);
124       }
125     }
126     List<E> list = new ArrayList<E>();
127     list.addAll(set);
128     logger.warn("List : " + list);
129     return list;
130   }
131 
132   private synchronized void commit() {
133     try {
134       odb.commit();
135     } catch (IOException e) {
136       logger.error("Exception comitting the ODB database", e);
137     } catch (Exception e) {
138       logger.error("Exception comitting the ODB database", e);
139     }
140   }
141 
142   /**
143    * {@inheritDoc}
144    */
145   @SuppressWarnings("unchecked")
146   public synchronized <E extends Composite<?, ?>> E persist(E composite) {
147     try {
148       setIds(composite);
149       E duplicate = (E) find(composite.getClass(), composite.getId());
150       if (duplicate != null) {
151         logger.warn("Attempted to persist a duplicate composite : " + composite);
152         return composite;
153       }
154       logger.debug("Persisting composite : " + composite);
155       odb.store(composite);
156     } catch (Exception e) {
157       logger.error("Exception persisting object : " + composite, e);
158     }
159     commit();
160     return composite;
161   }
162 
163   /**
164    * {@inheritDoc}
165    */
166   public synchronized <E extends Composite<?, ?>> E remove(Class<E> klass, Long id) {
167     E composite = find(klass, id);
168     try {
169       if (composite != null) {
170         odb.delete(composite);
171       }
172     } catch (Exception e) {
173       logger.error("Exception deleting object : " + id, e);
174     }
175     commit();
176     return composite;
177   }

179   /**
180    * {@inheritDoc}
181    */
182   public synchronized boolean isClosed() {
183     return closed;
184   }
185 
186   /**
187    * {@inheritDoc}
188    */
189   public synchronized void close() {
190     try {
191       if (closed) {
192         logger.warn("Attempted to close the database again");
193         return;
194       }
195       commit();
196       odb.close();
197       closed = true;
198 
199       final IDataBase dataBase = this;
200       IDataBaseEvent dataBaseEvent = new IDataBaseEvent() {
201         public IDataBase getDataBase() {
202           return dataBase;
203         }
204 
205         public Type getEventType() {
206           return IDataBaseEvent.Type.DATABASE_CLOSE;
207         }
208       };
209       dataBaseListener.fireDataBaseEvent(dataBaseEvent);
210     } catch (Exception e) {
211       logger.error("Exception closing the ODB database", e);
212     }
213   }
214 
215   @SuppressWarnings("unchecked")
216   private synchronized <E extends Composite<?, ?>> E find(IQuery query) {
217     E e = null;
218     try {
219       Objects objects = odb.getObjects(query);
220       if (objects.size() == 1) {
221         e = (E) objects.getFirst();
222       } else if (objects.size() > 1) {
223         logger.warn("Id for object must be unique : " + query);
224       }
225     } catch (Exception ex) {
226       logger.error("Exception selecting object on ODB database : " + query, ex);
227     }
228     return e;
229   }

231   /**
232    * This method sets the ids in a graph of objects.
233    * 
234    * @param object
235    *            the object to set the ids for
236    */
237   @SuppressWarnings("unchecked")
238   synchronized final void setIds(Composite<?, ?> composite) {
239     if (composite == null) {
240       return;
241     }
242     super.setId(composite);
243     logger.debug("Persisted object : " + composite);
244     List<Composite<?, ?>> children = (List<Composite<?, ?>>) composite.getChildren();
245     if (children != null) {
246       for (Composite<?, ?> child : children) {
247         setIds(child);
248       }
249     }
250   }
