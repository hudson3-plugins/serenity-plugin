080   /**
081    * This method collects the Java source for the class.
082    * 
083    * @param className
084    *            the name of the class
085    * @param source
086    *            the source for the class
087    */
088   public static final void collectSource(String className, String source) {
089     Class<Package<?, ?>, Method<?, ?>> klass = getClass(className);
090     File file = new File(IConstants.SERENITY_SOURCE, className + ".html");
091     if (!file.getParentFile().exists()) {
092       file.getParentFile().mkdirs();
093     }
094     if (!file.exists()) {
095       try {
096         file.createNewFile();
097       } catch (IOException e) {
098         LOGGER.error("Exception creating the source HTML file", e);
099       }
100     }
101     if (file.exists()) {
102       ISourceCode sourceCode = new CoverageSourceCode(klass, source);
103       String htmlSource = sourceCode.getSource();
104       Toolkit.setContents(file, htmlSource.getBytes());
105     }
106   }

128   /**
129    * Collects the packages that the class references and adds them to the document.
130    * 
131    * @param className
132    *            the name of the classes
133    * @param targetClassNames
134    *            the referenced class names
135    */
136   public static final void collectEfferentAndAfferent(String className, String... targetClassNames) {
137     String packageName = Toolkit.classNameToPackageName(className);
138     for (String targetClassName : targetClassNames) {
139       // Is the target name outside the package for this class
140       String targetPackageName = Toolkit.classNameToPackageName(targetClassName);
141       if (targetPackageName.trim().equals("")) {
142         continue;
143       }
144       // Is the target and the source the same package name
145       if (targetPackageName.equals(packageName)) {
146         continue;
147       }
148       // Exclude java.lang classes and packages
149       if (Configuration.getConfiguration().excluded(packageName) || Configuration.getConfiguration().excluded(targetPackageName)) {
150         continue;
151       }
152       // Add the target package name to the afferent packages for this package
153       Class<Package<?, ?>, Method<?, ?>> klass = getClass(className);
154       Afferent afferent = getAfferent(klass, targetPackageName);
155       if (!klass.getAfferent().contains(afferent)) {
156         klass.getAfferent().add(afferent);
157       }
158       // Add this package to the efferent packages of the target
159       Class<Package<?, ?>, Method<?, ?>> targetClass = getClass(targetClassName);
160       Efferent efferent = getEfferent(targetClass, packageName);
161       if (!targetClass.getEfferent().contains(efferent)) {
162         targetClass.getEfferent().add(efferent);
163       }
164     }
165   }

184   /**
185    * Collects the inner class for a class.
186    * 
187    * @param innerName
188    *            the name of the inner class
189    * @param outerName
190    *            the name of the outer class

192   public static final void collectInnerClass(String innerName, String outerName) {
193     Class<?, ?> innerClass = getClass(innerName);
194     Class<?, ?> outerClass = getClass(outerName);
195     if (innerClass.getOuterClass() == null) {
196       innerClass.setOuterClass(outerClass);
197     }
198     if (!outerClass.getInnerClasses().contains(innerClass)) {
199       outerClass.getInnerClasses().add(innerClass);
200     }
201   }

203   /**
204    * Collects the outer class of an inner class.
205    * 
206    * @param innerName
207    *            the name of the inner class
208    * @param outerName
209    *            the name of the outer class
210    * @param outerMethodName
211    *            the method name in the case this is an in-method class definition
212    * @param outerMethodDescription
213    *            the description of the method for anonymous and inline inner classes
214    */
215   public static final void collectOuterClass(String innerName, String outerName, String outerMethodName, String outerMethodDescription) {
216     Class<?, ?> innerClass = getClass(innerName);
217     Class<?, ?> outerClass = getClass(outerName);
218     if (innerClass.getOuterClass() == null) {
219       innerClass.setOuterClass(outerClass);
220     }
221     if (!outerClass.getInnerClasses().contains(innerClass)) {
222       outerClass.getInnerClasses().add(innerClass);
223     }
224     if (innerClass.getOuterMethod() == null) {
225       if (outerMethodName != null) {
226         Method<?, ?> outerMethod = getMethod(outerName, outerMethodName, outerMethodDescription);
227         innerClass.setOuterMethod(outerMethod);
228       }
229     }
230   }

232   @SuppressWarnings("unchecked")
233   private static final Package<Project<?, ?>, Class<?, ?>> getPackage(String className) {
234     String packageName = Toolkit.classNameToPackageName(className);
235 
236     long id = Toolkit.hash(packageName);
237     Package<Project<?, ?>, Class<?, ?>> pakkage = (Package<Project<?, ?>, Class<?, ?>>) dataBase.find(Package.class, id);
238 
239     if (pakkage == null) {
240       pakkage = new Package<Project<?, ?>, Class<?, ?>>();
241       pakkage.setName(packageName);
242       pakkage.setComplexity(1d);
243       pakkage.setCoverage(0d);
244       pakkage.setAbstractness(0d);
245       pakkage.setStability(0d);
246       pakkage.setDistance(0d);
247       pakkage.setInterfaces(0d);
248       pakkage.setImplementations(0d);
249       pakkage = (Package<Project<?, ?>, Class<?, ?>>) dataBase.persist(pakkage);
250 
251       LOGGER.debug("Added package : " + pakkage);
252     }
253     return pakkage;
254   }

256   @SuppressWarnings("unchecked")
257   private static final Class<Package<?, ?>, Method<?, ?>> getClass(String className) {
258     long id = Toolkit.hash(className);
259     Class klass = (Class) dataBase.find(Class.class, id);
260 
261     if (klass == null) {
262       klass = new Class();
263 
264       klass.setName(className);
265       klass.setComplexity(1d);
266       klass.setCoverage(0d);
267       klass.setStability(0d);
268       klass.setEfference(0d);
269       klass.setAfference(0d);
270       klass.setInterfaze(false);
271 
272       Package<Project<?, ?>, Class<?, ?>> pakkage = getPackage(className);
273       pakkage.getChildren().add(klass);
274       klass.setParent(pakkage);
275 
276       klass = (Class<?, ?>) dataBase.persist(klass);
277       LOGGER.debug("Added class  : " + klass);
278     }
279     return klass;
280   }

282   @SuppressWarnings("unchecked")
283   private static final Method<?, ?> getMethod(String className, String methodName, String methodDescription) {
284     methodName = methodName.replace('<', ' ').replace('>', ' ').trim();
285 
286     long id = Toolkit.hash(className, methodName, methodDescription);
287     Method method = (Method) dataBase.find(Method.class, id);
288 
289     if (method == null) {
290       method = new Method();
291 
292       method.setName(methodName);
293       method.setClassName(className);
294       method.setDescription(methodDescription);
295       method.setComplexity(0d);
296       method.setCoverage(0d);
297 
298       Class klass = getClass(className);
299       method.setParent(klass);
300       if (klass.getChildren() == null) {
301         List<Method> children = new ArrayList<Method>();
302         klass.setChildren(children);
303       }
304       klass.getChildren().add(method);
305 
306       dataBase.persist(method);
307     }
308     return method;
309   }

311   @SuppressWarnings("unchecked")
312   protected static final Line<?, ?> getLine(String className, String methodName, String methodDescription, double lineNumber) {
313     Line line = null;
314     long id = Toolkit.hash(className, methodName, lineNumber);
315     line = (Line) dataBase.find(Line.class, id);
316 
317     if (line == null) {
318       line = new Line();
319 
320       line.setNumber(lineNumber);
321       line.setCounter(0d);
322       line.setClassName(className);
323       line.setMethodName(methodName);
324 
325       Method method = getMethod(className, methodName, methodDescription);
326       Collection<Composite> lines = method.getChildren();
327       line.setParent(method);
328       lines.add(line);
329 
330       dataBase.persist(line);
331     }
332     return line;
333   }

335   private static final Efferent getEfferent(Class<?, ?> klass, String packageName) {
336     StringBuilder builder = new StringBuilder("<e:");
337     builder.append(packageName);
338     builder.append(">");
339     String name = builder.toString();
340 
341     long id = Toolkit.hash(name);
342     Efferent efferent = (Efferent) dataBase.find(Efferent.class, id);
343     if (efferent == null) {
344       efferent = new Efferent();
345       efferent.setName(name);
346 
347       klass.getEfferent().add(efferent);
348 
349       dataBase.persist(efferent);
350     }
351     return efferent;
352   }
