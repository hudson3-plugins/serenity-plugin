018 /**
019  * Just some useful methods to dump the database and clean it.
020  * 
021  * @author Michael Couck
022  * @since 09.12.09
023  * @version 01.00
024  */
025 public class DataBaseToolkit {
026 
027   static {
028     LoggingConfigurator.configure();
029   }
030   private static Logger logger = Logger.getLogger(DataBaseToolkit.class);
031 
032   /**
033    * Clears the data in the database.
034    * 
035    * @param dataBase
036    *            the database to truncate
037    */
038   @SuppressWarnings("unchecked")
039   public static void clear(IDataBase dataBase) {
040     Project<?, ?> project = (Project<?, ?>) dataBase.find(Project.class, Toolkit.hash(Project.class.getName()));
041     if (project != null) {
042       dataBase.remove(Project.class, project.getId());
043     }
044     List<Package> packages = dataBase.find(Package.class);
045     for (Composite<?, ?> composite : packages) {
046       dataBase.remove(composite.getClass(), composite.getId());
047     }
048     List<Class> classes = dataBase.find(Class.class);
049     for (Composite composite : classes) {
050       dataBase.remove(composite.getClass(), composite.getId());
051     }
052     List<Method> methods = dataBase.find(Method.class);
053     for (Composite composite : methods) {
054       dataBase.remove(composite.getClass(), composite.getId());
055     }
056     List<Line> lines = dataBase.find(Line.class);
057     for (Composite composite : lines) {
058       dataBase.remove(composite.getClass(), composite.getId());
059     }
060   }
061 
062   /**
063    * Dumps the database to the output stream.
064    * 
065    * @param dataBase
066    *            the database to dump
067    * @param criteria
068    *            the criteria to match if the data for the composite must be written to the output
069    */
070   @SuppressWarnings("unchecked")
071   public static void dump(IDataBase dataBase, ICriteria criteria, String message) {
072     if (message != null) {
073       logger.warn(message);
074     }
075     try {
076       Object object = dataBase.find(Project.class, Toolkit.hash(Project.class.getName()));
077       logger.info(object);
078       Project<?, ?> project = (Project<?, ?>) dataBase.find(Project.class, Toolkit.hash(Project.class.getName()));
079       if (project != null) {
080         logger.warn("Project : " + project.getName());
081       }
082     } catch (Exception e) {
083       logger.error("Exception dumping the data for the project object.", e);
084     }
085     try {
086       List<Package> packages = dataBase.find(Package.class);
087       for (Package<?, ?> pakkage : packages) {
088         log(criteria, pakkage, 1, pakkage.getId() + " : " + pakkage.getName() + ", coverage : " + pakkage.getCoverage() + ", complexity : "
089             + pakkage.getComplexity() + ", stability : " + pakkage.getStability());
090         for (Class<?, ?> klass : ((List<Class<?, ?>>) pakkage.getChildren())) {
091           log(criteria, klass, 2, " : id : " + klass.getId() + " : name : " + klass.getName() + " : coverage : " + klass.getCoverage()
092               + ", complexity : " + klass.getComplexity() + ", outer class : " + klass.getOuterClass() + ", outer method : "
093               + klass.getOuterMethod() + ", lines : " + klass.getChildren().size() + ", inner classes : " + klass.getInnerClasses());
094           List<Efferent> efferents = klass.getEfferent();
095           List<Afferent> afferents = klass.getAfferent();
096           for (Efferent efferent : efferents) {
097             log(criteria, efferent, 4, efferent.getName());
098           }
099           for (Afferent afferent : afferents) {
100             log(criteria, afferent, 4, afferent.getName());
101           }
102           for (Method<?, ?> method : ((List<Method<?, ?>>) klass.getChildren())) {
103             log(criteria, method, 3, method.getId() + " : name : " + method.getName() + " : coverage : " + method.getCoverage()
104                 + ", complexity : " + method.getComplexity());
105             for (Line<?, ?> line : ((List<Line<?, ?>>) method.getChildren())) {
106               log(criteria, line, 4, line.getId() + " : number : " + line.getNumber() + ", counter : " + line.getCounter());
107             }
108           }
109         }
110       }
111     } catch (Exception e) {
112       logger.error("Exception dumping the data for the database.", e);
113     }
114   }

116   private static void log(ICriteria criteria, Composite<?, ?> composite, int tabs, String data) {
117     if (criteria == null || (criteria != null && criteria.satisfied(composite))) {
118       StringBuilder builder = new StringBuilder();
119       for (int i = 0; i < tabs; i++) {
120         builder.append("\t");
121       }
122       builder.append(composite.getClass().getSimpleName());
123       builder.append(" : ");
124       builder.append(data);
125       logger.warn(builder.toString());
126     }
127   }

131     public boolean satisfied(Composite<?, ?> composite);
132 
133   }
