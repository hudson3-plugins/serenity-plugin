01 package com.ikokoon.serenity.instrumentation.coverage;
02 
03 import org.apache.log4j.Logger;
04 import org.objectweb.asm.Label;
05 import org.objectweb.asm.MethodAdapter;
06 import org.objectweb.asm.MethodVisitor;
07 import org.objectweb.asm.Opcodes;
08 import org.objectweb.asm.Type;
09 
10 import com.ikokoon.serenity.Collector;
11 import com.ikokoon.toolkit.Toolkit;
12 
13 /**
14  * This class actually enhances the lines to call the collector class which gathers the data on the lines that are executed during the unit tests.
15  * 
16  * @author Michael Couck
17  * @since 12.07.09
18  * @version 01.00
19  */
20 public class CoverageMethodAdapter extends MethodAdapter {
21 
22   /** The logger for the class. */
23   private Logger logger = Logger.getLogger(CoverageMethodAdapter.class);
24 
25   /** The type of parameters that the {@link Collector} takes in the coverage collection method. */
26   private Type stringType = Type.getType(String.class);
27   /** The type parameter for the line number in the {@link Collector} collect coverage method. */
28   private Type intType = Type.getType(int.class);
29   /** The array of type parameters for the {@link Collector} for the coverage method. */
30   private Type[] types = new Type[] { stringType, stringType, stringType, intType };
31 
32   /** The name of the class ({@link Collector}) that will be the collector for the method adapter. */
33   private String collectorClassName = Type.getInternalName(Collector.class);
34   /** The coverage method that is called on the {@link Collector} by the added instructions. */
35   private String collectorMethodName = "collectCoverage";
36   /** The byte code signature of the coverage method in the {@link Collector}. */
37   private String collectorMethodDescription = Type.getMethodDescriptor(Type.VOID_TYPE, types);
38 
39   /** The name of the class that this method adapter is enhancing the methods for. */
40   private String className;
41   /** The name of the method that is being enhanced. */
42   private String methodName;
43   /** The description of the method being enhanced. */
44   private String methodDescription;
45 
46   /**
47    * The constructor initialises a {@link CoverageMethodAdapter} that takes all the interesting items for the method that is to be enhanced
48    * including the parent method visitor.
49    * 
50    * @param methodVisitor
51    *            the method visitor of the parent
52    * @param className
53    *            the name of the class the method belongs to
54    * @param methodName
55    *            the name of the method
56    * @param methodDescription
57    *            the description of the method
58    */
59   public CoverageMethodAdapter(MethodVisitor methodVisitor, String className, String methodName, String methodDescription) {
60     super(methodVisitor);
61     this.className = Toolkit.slashToDot(className);
62     this.methodName = methodName;
63     this.methodDescription = methodDescription;
64     logger.debug("Class name : " + className + ", name : " + methodName + ", desc : " + methodDescription);
65   }
66 
67   /**
68    * This is the method that actually adds the instructions to the enhanced class. It adds an instruction to call a collector class which then
69    * collects the data about each line being called. This method puts five strings onto the stack. These are then popped by the call to the
70    * collector class and passed as parameters to the collector method.
71    */
72   public void visitLineNumber(int lineNumber, Label label) {
73     logger.debug("visitLineNumber : " + lineNumber + ", " + label + ", " + label.getOffset() + ", " + className + ", " + methodName);
74     this.mv.visitLdcInsn(className);
75     this.mv.visitLdcInsn(methodName);
76     this.mv.visitLdcInsn(methodDescription);
77     this.mv.visitLdcInsn(lineNumber);
78     this.mv.visitMethodInsn(Opcodes.INVOKESTATIC, collectorClassName, collectorMethodName, collectorMethodDescription);
79     this.mv.visitLineNumber(lineNumber, label);
80   }
81 
82 }
