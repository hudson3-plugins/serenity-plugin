01 package com.ikokoon.serenity.instrumentation.complexity;
02 
03 import java.util.Arrays;
04 
05 import org.apache.log4j.Logger;
06 import org.objectweb.asm.ClassAdapter;
07 import org.objectweb.asm.ClassVisitor;
08 import org.objectweb.asm.MethodAdapter;
09 import org.objectweb.asm.MethodVisitor;
10 
11 import com.ikokoon.serenity.instrumentation.VisitorFactory;
12 import com.ikokoon.toolkit.Toolkit;
13 
14 /**
15  * This is the top level class adapter for collecting the complexity for the classes. It just calls the complexity method adapter where the real work
16  * happens.
17  * 
18  * @author Michael Couck
19  * @since 12.07.09
20  * @version 01.00
21  */
22 public class ComplexityClassAdapter extends ClassAdapter {
23 
24   private Logger logger = Logger.getLogger(this.getClass());
25   /** The name of the class that is being instrumented. */
26   private String className;
27 
28   /**
29    * Constructor initialises a {@link ComplexityClasAdapter} ComplexityClassAdapter with the parent visitor and the name of the class that will be
30    * parsed for complexity or jump instructions.
31    * 
32    * @param visitor
33    *            the parent visitor what will be called to collect the byte code
34    * @param className
35    *            the name of the class being parsed
36    */
37   public ComplexityClassAdapter(ClassVisitor visitor, String className) {
38     super(visitor);
39     this.className = Toolkit.slashToDot(className);
40     logger.debug("Constructor : " + className);
41   }
42 
43   /**
44    * {@inheritDoc}
45    */
46   public void visit(int version, int access, String className, String signature, String superName, String[] interfaces) {
47     if (logger.isDebugEnabled()) {
48       logger.debug("visit : " + version + ", " + access + ", " + className + ", " + signature + ", " + superName);
49       if (interfaces != null) {
50         logger.debug(Arrays.asList(interfaces).toString());
51       }
52     }
53     super.visit(version, access, className, signature, superName, interfaces);
54   }
55 
56   /**
57    * {@inheritDoc}
58    */
59   public MethodVisitor visitMethod(int access, String methodName, String methodDescription, String signature, String[] exceptions) {
60     logger.debug("visitMethod : " + access + ", " + methodName + ", " + methodDescription + ", " + signature + ", " + exceptions);
61     MethodVisitor methodVisitor = super.visitMethod(access, methodName, methodDescription, signature, exceptions);
62     MethodAdapter methodAdapter = (MethodAdapter) VisitorFactory.getMethodVisitor(methodVisitor, ComplexityMethodAdapter.class, className,
63         methodName, methodDescription);
64     return methodAdapter;
65   }
66 
67 }
