01 package com.ikokoon.serenity.hudson;
02 
03 import hudson.Plugin;
04 
05 import java.util.Map;
06 
07 import org.apache.log4j.Logger;
08 
09 import com.ikokoon.serenity.LoggingConfigurator;
10 import com.ikokoon.serenity.persistence.IDataBase;
11 
12 /**
13  * Entry point of a plugin.
14  * 
15  * <p>
16  * There must be one class in each plugin. Actually not any more. If there is no plugin in the plugin then Hudson will create one it seems.. See
17  * JavaDoc of for more about what can be done on this class.
18  * 
19  * @plugin
20  * 
21  * @author Michael Couck
22  * @since 25.07.09
23  * @version 01.00
24  */
25 public class SerenityPlugin extends Plugin {
26 
27   /** The logger for the plugin class. */
28   private Logger logger;
29 
30   /**
31    * Constructor initialises the logging and the database.
32    */
33   public SerenityPlugin() {
34     LoggingConfigurator.configure();
35     logger = Logger.getLogger(SerenityPlugin.class);
36     logger.info("Loaded plugin : " + this.getClass().getName());
37   }
38 
39   /**
40    * We need to stop all the databases. This releases memory and allows the databases to be committed in the case any objects were modified, which
41    * generally they shouldn't be of course.
42    */
43   @Override
44   public void stop() {
45     Map<String, IDataBase> dataBases = IDataBase.DataBaseManager.getDataBases();
46     IDataBase[] dataBasesArray = dataBases.values().toArray(new IDataBase[dataBases.values().size()]);
47     for (IDataBase dataBase : dataBasesArray) {
48       dataBase.close();
49     }
50   }
51 
52 }
