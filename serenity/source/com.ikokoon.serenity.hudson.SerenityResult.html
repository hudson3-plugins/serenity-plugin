056   /**
057    * This method is called from the front end. The result from the call will result in some piece of data being extracted from the database. For
058    * example if the user clicks on a package the name of the package will be used to get that package from the database and will be made available
059    * to the UI.
060    * 
061    * @param token
062    *            the token from the front end, could be a package name or a class name
063    * @param req
064    *            the Stapler request from the ui
065    * @param rsp
066    *            the Stapler response for the ui
067    * @return the result which is this class
068    * @throws IOException
069    */
070   @SuppressWarnings("unchecked")
071   public Object getDynamic(String token, StaplerRequest req, StaplerResponse rsp) throws Exception {
072     logger.debug("getDynamic:" + token);
073 
074     // printParameters(req);
075 
076     String klass = req.getParameter("class");
077     String id = req.getParameter("id");
078 
079     IDataBase dataBase = null;
080     try {
081       if (klass != null && id != null) {
082         dataBase = getDataBase(abstractBuild);
083         composite = dataBase.find((java.lang.Class<Composite<?, ?>>) java.lang.Class.forName(klass), Long.parseLong(id));
084         logger.debug("Class : " + klass + ", id : " + id + ", " + composite);
085       }
086     } catch (Exception e) {
087       logger.error("Exception initialising the model and the source for : " + klass + ", " + id, e);
088     } finally {
089       closeDataBase(dataBase);
090     }
091 
092     return this;
093   }

133   @SuppressWarnings("unchecked")
134   public List<Package> getPackages() {
135     logger.debug("getPackages");
136     IDataBase dataBase = null;
137     try {
138       dataBase = getDataBase(abstractBuild);
139       List<Package> packages = dataBase.find(Package.class);
140       if (packages != null) {
141         Collections.sort(packages, new Comparator<Package>() {
142           public int compare(Package o1, Package o2) {
143             return o1.getName().compareTo(o2.getName());
144           }
145         });
146         // Sort the classes in the packages
147         for (Package<?, ?> pakkage : packages) {
148           Collections.sort(pakkage.getChildren(), new Comparator<Class<?, ?>>() {
149             public int compare(Class<?, ?> o1, Class<?, ?> o2) {
150               return o1.getName().compareTo(o2.getName());
151             }
152           });
153         }
154       }
155       // Remove the inner classes from the packages, and set the precision
156       for (Package<?, ?> pakkage : packages) {
157         Iterator<Class<?, ?>> iterator = pakkage.getChildren().iterator();
158         while (iterator.hasNext()) {
159           Class<?, ?> klass = iterator.next();
160           if (klass.getName().indexOf("$") > -1) {
161             iterator.remove();
162           }
163         }
164       }
165       return packages;
166     } finally {
167       closeDataBase(dataBase);
168     }
169   }

176   public String getProjectModel() {
177     IDataBase dataBase = getDataBase(abstractBuild);
178     Project<?, ?> project = dataBase.find(Project.class, Toolkit.hash(Project.class.getName()));
179     // Move the build forward to the last build because Hudson will go to the last stable build
180     // which we don't want, we want the last build
181     abstractBuild = getLastBuild(abstractBuild);
182 
183     Object object = abstractBuild.getProject();
184     if (object instanceof hudson.model.Project) {
185       hudson.model.Project<?, ?> hudsonProject = (hudson.model.Project<?, ?>) object;
186       String projectName = hudsonProject.getName();
187       project.setName(projectName);
188     }
189 
190     String model = getModel(project, "highchartsSmall.model");
191     return model;
192   }

210   private String getSource(Composite<?, ?> composite) {
211     logger.debug("getSource");
212     if (composite instanceof Class) {
213       String className = ((Class<?, ?>) composite).getName();
214       String sourceFilePath = abstractBuild.getRootDir().getAbsolutePath() + File.separator + IConstants.SERENITY_SOURCE + File.separator
215           + className + ".html";
216       logger.debug("Looking for source : " + sourceFilePath);
217       File sourceFile = new File(sourceFilePath);
218       if (sourceFile.exists()) {
219         String source = Toolkit.getContents(sourceFile).toString();
220         return source;
221       }
222     }
223     return "No source";
224   }

226   @SuppressWarnings("unchecked")
227   public String getModel(Composite<?, ?> composite, String modelName) {
228     logger.debug("getModel");
229     if (composite == null) {
230       return "";
231     }
232     LinkedList<Composite<?, ?>> composites = new LinkedList<Composite<?, ?>>();
233     composites.addFirst(composite);
234     LinkedList<Integer> buildNumbers = new LinkedList<Integer>();
235     buildNumbers.addFirst(abstractBuild.number);
236 
237     composites = getPreviousComposites((java.lang.Class<Composite<?, ?>>) composite.getClass(), abstractBuild, composites, buildNumbers,
238         composite.getId(), 1);
239 
240     IModeller modeller = new HighchartsModeller(modelName, buildNumbers.toArray(new Integer[buildNumbers.size()]));
241     modeller.visit(composite.getClass(), composites.toArray(new Composite[composites.size()]));
242     String model = modeller.getModel();
243     return model;
244   }

246   @SuppressWarnings("unchecked")
247   LinkedList<Composite<?, ?>> getPreviousComposites(java.lang.Class<Composite<?, ?>> klass, AbstractBuild<?, ?> abstractBuild,
248       LinkedList<Composite<?, ?>> composites, LinkedList<Integer> buildNumbers, Long id, int history) {
249     if (history >= HISTORY) {
250       return composites;
251     }
252     logger.debug("Abstract build : " + abstractBuild);
253     AbstractBuild<?, ?> previousBuild = abstractBuild.getPreviousBuild();
254 
255     if (previousBuild == null) {
256       return composites;
257     }
258     IDataBase dataBase = getDataBase(previousBuild);
259     Composite<?, ?> composite = dataBase.find(klass, id);
260     logger.debug("Looking for composite : " + id + ", " + composite);
261     dataBase.close();
262     if (composite == null) {
263       return composites;
264     }
265     composites.addFirst(composite);
266     buildNumbers.addFirst(previousBuild.number);
267     return getPreviousComposites((java.lang.Class<Composite<?, ?>>) composite.getClass(), previousBuild, composites, buildNumbers, id, ++history);
268   }
