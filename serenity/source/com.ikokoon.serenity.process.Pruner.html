01 package com.ikokoon.serenity.process;
02 
03 import java.util.List;
04 
05 import com.ikokoon.serenity.IConstants;
06 import com.ikokoon.serenity.model.Class;
07 import com.ikokoon.serenity.model.Line;
08 import com.ikokoon.serenity.model.Method;
09 import com.ikokoon.serenity.model.Package;
10 import com.ikokoon.serenity.persistence.IDataBase;
11 
12 /**
13  * This class removes the methods and lines from the model as we will not need them further and they form a very large part of the model which hogs
14  * memory.
15  * 
16  * @author Michael Couck
17  * @since 10.01.10
18  * @version 01.00
19  */
20 public class Pruner extends AProcess implements IConstants {
21 
22   /** The database to prune. */
23   private IDataBase dataBase;
24 
25   /**
26    * Constructor takes the parent.
27    * 
28    * @param parent
29    *            the parent process that will chain this process
30    */
31   public Pruner(IProcess parent, IDataBase dataBase) {
32     super(parent);
33     this.dataBase = dataBase;
34   }
35 
36   /**
37    * {@inheritDoc}
38    */
39   @SuppressWarnings("unchecked")
40   public void execute() {
41     super.execute();
42     List<Package> packages = dataBase.find(Package.class);
43     for (Package<?, ?> pakkage : packages.toArray(new Package[packages.size()])) {
44       for (Class<?, ?> klass : pakkage.getChildren()) {
45         for (Method<?, ?> method : klass.getChildren()) {
46           for (Line<?, ?> line : method.getChildren()) {
47             line.setParent(null);
48           }
49           method.setParent(null);
50           method.setChildren(null);
51         }
52         klass.setAfferent(null);
53         klass.setEfferent(null);
54         klass.setChildren(null);
55       }
56       pakkage.setAfferent(null);
57       pakkage.setEfferent(null);
58       pakkage.setParent(null);
59     }
60   }
61 
62 }
