004 import java.util.Map;
005 
006 import javax.persistence.EntityManager;
007 import javax.persistence.EntityManagerFactory;
008 import javax.persistence.EntityTransaction;
009 import javax.persistence.Persistence;
010 import javax.persistence.PersistenceContext;
011 import javax.persistence.PersistenceContextType;

013 import org.apache.log4j.Logger;
014 
015 import com.ikokoon.serenity.IConstants;

019 /**
020  * This is the JPA database implementation. This class could be used in the future if so desired by a user of the Serenity plugin. The issue with the
021  * JPA database is the speed of persistence and where the database will be. This could be configurable of course, but then the user has to do some
022  * serious configuration which will require technical knowledge, and of course set up a database. Too much effort I thinks.
023  * 
024  * @author Michael Couck
025  * @since 20.12.09
026  * @version 01.00
027  */
028 public class DataBaseJpa extends DataBase {
029 
030   private Logger logger = Logger.getLogger(this.getClass());
031   @PersistenceContext(type = PersistenceContextType.TRANSACTION, unitName = IConstants.SERENITY_PERSISTENCE_UNIT)
032   private EntityManager entityManager;
033   private boolean closed = true;
034 
035   /**
036    * Constructor opens the entity manager.
037    */
038   DataBaseJpa() {
039     getEntityManager();
040     closed = false;
041   }
042 
043   /**
044    * {@inheritDoc}
045    */
046   public synchronized <E extends Composite<?, ?>> E find(Class<E> klass, Long id) {
047     boolean commit = begin();
048     try {
049       E e = entityManager.find(klass, id);
050       logger.info("find(klass, id) : " + e);
051       return e;
052     } catch (Exception e) {
053       logger.error("Exception finding : " + klass + ", " + id, e);
054     } finally {
055       commit(commit);
056     }
057     return null;
058   }
059 
060   public synchronized <E extends Composite<?, ?>> E find(Class<E> klass, List<Object> parameters) {
061     Long id = Toolkit.hash(parameters.toArray());
062     return find(klass, id);
063   }
064 
065   /**
066    * {@inheritDoc}
067    */
068   @SuppressWarnings("unchecked")
069   public synchronized <E extends Composite<?, ?>> List<E> find(Class<E> klass) {
070     return entityManager.createQuery("from " + klass.getSimpleName()).getResultList();
071   }
072 
073   /**
074    * {@inheritDoc}
075    */
076   public <E extends Composite<?, ?>> List<E> find(Class<E> klass, Map<String, Object> parameters) {
077     throw new RuntimeException("Not implempented.");
078   }
079 
080   /**
081    * {@inheritDoc}
082    */
083   public synchronized boolean isClosed() {
084     return closed;
085   }
086 
087   /**
088    * {@inheritDoc}
089    */
090   public synchronized <E extends Composite<?, ?>> E persist(E composite) {
091     // setId(composite);
092     boolean commit = begin();
093     entityManager.persist(composite);
094     // entityManager.refresh(composite);
095     commit(commit);
096     return composite;
097   }
098 
099   /**
100    * {@inheritDoc}
101    */
102   public synchronized <E extends Composite<?, ?>> E remove(Class<E> klass, Long id) {
103     E composite = find(klass, id);
104     if (composite != null) {
105       boolean commit = begin();
106       entityManager.remove(composite);
107       commit(commit);
108     }
109     return composite;
110   }
111 
112   /**
113    * {@inheritDoc}
114    */
115   public synchronized void close() {
116     logger.info("Closing the JPA database : " + entityManager.isOpen());
117     if (entityManager.isOpen()) {
118       boolean commit = begin();
119       try {
120         entityManager.flush();
121       } finally {
122         commit(commit);
123       }
124       entityManager.close();
125       closed = true;
126     }
127   }
128 
129   /**
130    * This method starts a transaction for the entity manager.
131    */
132   private synchronized boolean begin() {
133     EntityTransaction transaction = entityManager.getTransaction();
134     if (!transaction.isActive()) {
135       entityManager.getTransaction().begin();
136       return true;
137     }
138     return false;
139   }
140 
141   /**
142    * This method commits the transaction for he entity manager.
143    */
144   private synchronized void commit(boolean commit) {
145     if (commit) {
146       EntityTransaction transaction = entityManager.getTransaction();
147       if (transaction.isActive()) {
148         if (!transaction.getRollbackOnly()) {
149           transaction.commit();
150         } else {
151           transaction.rollback();
152         }
153       }
154     }
155   }
156 
157   /**
158    * Initialises the entity manager.
159    */
160   private synchronized EntityManager getEntityManager() {
161     if (entityManager == null) {
162       try {
163         EntityManagerFactory factory = Persistence.createEntityManagerFactory(IConstants.SERENITY_PERSISTENCE_UNIT);
164         entityManager = factory.createEntityManager();
165       } catch (Exception e) {
166         logger.error("Error setting up the entity manager in stand alone.", e);
167       }
168     }
169     return entityManager;
170   }
