01 package com.ikokoon.serenity.instrumentation.coverage;
02 
03 import org.apache.log4j.Logger;
04 import org.objectweb.asm.ClassAdapter;
05 import org.objectweb.asm.ClassVisitor;
06 import org.objectweb.asm.MethodAdapter;
07 import org.objectweb.asm.MethodVisitor;
08 import org.objectweb.asm.Opcodes;
09 
10 import com.ikokoon.serenity.instrumentation.VisitorFactory;
11 import com.ikokoon.toolkit.Toolkit;
12 
13 /**
14  * This is the class visitor that visits the class structures and invokes the method visitor for the coverage functionality.
15  * 
16  * @author Michael Couck
17  * @since 12.07.09
18  * @version 01.00
19  */
20 public class CoverageClassAdapter extends ClassAdapter implements Opcodes {
21 
22   private Logger logger = Logger.getLogger(this.getClass());
23   /** The name of the class that is being instrumented. */
24   private String className;
25 
26   /**
27    * Constructs a {@link CoverageClassAdapter} that takes the parent class visitor and the class name that will be enhanced with instructions to
28    * invoke the {@link Collector} that collects the instructions that are executed.
29    * 
30    * @param visitor
31    *            the parent class visitor
32    * @param className
33    *            that name of the class that will be enhanced with coverage instructions
34    */
35   public CoverageClassAdapter(ClassVisitor visitor, String className) {
36     super(visitor);
37     this.className = Toolkit.slashToDot(className);
38     logger.debug("Constructor : " + className);
39   }
40 
41   /**
42    * This is the method that calls the MethodAdapter that will enhance the class methods with instructions that will enable data to be collected for
43    * the class at runtime producing a line coverage report for the class.
44    */
45   public MethodVisitor visitMethod(int access, String methodName, String methodDescription, String methodSignature, String[] exceptions) {
46     logger.debug("visitMethod : " + access + ", " + methodName + ", " + methodDescription + ", " + methodSignature + ", " + exceptions);
47     MethodVisitor methodVisitor = super.visitMethod(access, methodName, methodDescription, methodSignature, exceptions);
48     MethodAdapter methodAdapter = (MethodAdapter) VisitorFactory.getMethodVisitor(methodVisitor, CoverageMethodAdapter.class, className,
49         methodName, methodDescription);
50     return methodAdapter;
51   }
52 
53 }
