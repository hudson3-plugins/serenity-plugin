032 /**
033  * This class runs at the end of the build, called by Hudson. The purpose is to copy the database file from the output directory for the reports
034  * directory where the build action can present the data to the front end. As well as this the source that was found for the project is copied to the
035  * source directory where the front end can access it.
036  * 
037  * @author Michael Couck
038  * @since 12.07.09
039  * @version 01.00
040  */

042 public class SerenityPublisher extends Recorder {

056   /**
057    * {@inheritDoc}
058    */
059   @Override
060   public boolean perform(AbstractBuild<?, ?> build, Launcher launcher, BuildListener buildListener) throws InterruptedException, IOException {
061     logger.debug("perform");
062 
063     if (!Result.SUCCESS.equals(build.getResult())) {
064       buildListener.getLogger().println("Build was not successful... but will still try to publish the report");
065     }
066 
067     buildListener.getLogger().println("Publishing Serenity reports...");
068 
069     copyDataBaseToBuildDirectory(build, buildListener);
070     copySourceToBuildDirectory(build, buildListener);
071 
072     buildListener.getLogger().println("Accessing Serenity results...");
073     ISerenityResult result = new SerenityResult(build);
074     SerenityBuildAction buildAction = new SerenityBuildAction(build, result);
075     build.getActions().add(buildAction);
076 
077     return true;
078   }

080   private boolean copySourceToBuildDirectory(AbstractBuild<?, ?> build, final BuildListener buildListener) throws InterruptedException, IOException {
081     FilePath moduleRoot = build.getWorkspace();
082     buildListener.getLogger().println("Module root : " + moduleRoot.toURI().getRawPath());
083 
084     List<File> sourceDirectories = new ArrayList<File>();
085     Toolkit.findFiles(new File(moduleRoot.toURI().getRawPath()), new Toolkit.IFileFilter() {
086       public boolean matches(File file) {
087         if (file == null) {
088           return false;
089         }
090         if (!file.isDirectory()) {
091           return false;
092         }
093         String filePath = file.getAbsolutePath();
094         filePath = Toolkit.replaceAll(filePath, "\\", "/");
095         if (filePath.indexOf("serenity/source") == -1) {
096           return false;
097         }
098         // Exclude the Subversion directories if there are any
099         String[] excludedDirectories = { ".svn", "prop-base", "props", "text-base", "tmp" };
100         for (String excludedDirectory : excludedDirectories) {
101           if (filePath.indexOf(excludedDirectory) > -1) {
102             return false;
103           }
104         }
105         return true;
106       }
107     }, sourceDirectories);
108 
109     FilePath buildDirectory = new FilePath(build.getRootDir());
110     FilePath buildSourceDirectory = new FilePath(buildDirectory, IConstants.SERENITY_SOURCE);
111     File targetSourceDirectory = new File(buildSourceDirectory.toURI().getRawPath());
112 
113     try {
114       for (File sourceDirectory : sourceDirectories) {
115         buildListener.getLogger().println(
116             "Publishing serenity source from : " + sourceDirectory.getAbsolutePath() + ", to : "
117                 + buildSourceDirectory.toURI().getRawPath());
118         Toolkit.copyFile(sourceDirectory, targetSourceDirectory);
119       }
120     } catch (IOException e) {
121       Util.displayIOException(e, buildListener);
122       e.printStackTrace(buildListener.fatalError("Unable to copy Serenity database file from : " + sourceDirectories + ", to : "
123           + buildDirectory));
124     }
125     return true;
126   }

128   private boolean copyDataBaseToBuildDirectory(AbstractBuild<?, ?> build, BuildListener buildListener) throws InterruptedException, IOException {
129     String serenityDatabase = "**/serenity/serenity.odb";
130     FilePath[] reports = new FilePath[0];
131     final FilePath moduleRoot = build.getWorkspace();
132     buildListener.getLogger().println("Module root : " + moduleRoot.toURI().getRawPath());
133     try {
134       reports = moduleRoot.list(serenityDatabase);
135     } catch (IOException e) {
136       Util.displayIOException(e, buildListener);
137       e.printStackTrace(buildListener.fatalError("Unable to find Serenity results"));
138       build.setResult(Result.FAILURE);
139       return false;
140     }
141 
142     if (reports.length == 0) {
143       buildListener.getLogger().println("No coverage results were found using the pattern '" + serenityDatabase + "'.");
144       build.setResult(Result.FAILURE);
145       return true;
146     }
147 
148     if (reports.length > 1) {
149       buildListener.getLogger().println("Serenity publisher found more than one report that match the pattern. Impossible, but true.");
150       build.setResult(Result.FAILURE);
151       return true;
152     }
153 
154     FilePath buildTarget = new FilePath(build.getRootDir());
155     FilePath singleReport = reports[0];
156     FilePath targetPath = new FilePath(buildTarget, IConstants.DATABASE_FILE_ODB);
157 
158     try {
159       buildListener.getLogger().println(
160           "Publishing serenity db from : " + singleReport.toURI().getRawPath() + ", to : " + targetPath.toURI().getRawPath());
161       singleReport.copyTo(targetPath);
162     } catch (IOException e) {
163       Util.displayIOException(e, buildListener);
164       e.printStackTrace(buildListener.fatalError("Unable to copy Serenity database file from : " + singleReport + ", to : " + buildTarget));
165       build.setResult(Result.FAILURE);
166     }
167     return true;
168   }
