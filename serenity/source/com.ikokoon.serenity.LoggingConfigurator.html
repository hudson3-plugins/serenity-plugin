01 package com.ikokoon.serenity;
02 
03 import java.io.File;
04 import java.io.IOException;
05 import java.net.URL;
06 import java.util.Properties;
07 
08 import org.apache.log4j.PropertyConfigurator;
09 
10 /**
11  * This is a central class for initialising the Log4j logging parameters.
12  * 
13  * @author Michael Couck
14  * @since 09.12.09
15  * @version 01.00
16  */
17 public class LoggingConfigurator {
18 
19   /** Whether the logging has already be initialised. */
20   private static boolean initilised = false;
21   private static final String SERENITY_LOG_FILE = "./serenity/serenity.log";
22 
23   static {
24     configure();
25   }
26 
27   /**
28    * Configures the Jog4j properties. In the case the log4j.properties file is not on the classpath in the META-INF directory hard coded properties
29    * will be used.
30    */
31   public static void configure() {
32     if (!initilised) {
33       checkLogFolder(SERENITY_LOG_FILE);
34       URL url = LoggingConfigurator.class.getResource(IConstants.LOG_4_J_PROPERTIES);
35       if (url != null) {
36         PropertyConfigurator.configure(url);
37       } else {
38         Properties properties = getProperties();
39         PropertyConfigurator.configure(properties);
40       }
41       initilised = true;
42     }
43   }
44 
45   private static void checkLogFolder(String filePath) {
46     File file = new File(filePath);
47     if (!file.getParentFile().exists()) {
48       file.getParentFile().mkdirs();
49       try {
50         file.createNewFile();
51       } catch (IOException e) {
52         e.printStackTrace();
53       }
54     }
55   }
56 
57   private static Properties getProperties() {
58     Properties properties = new Properties();
59     // Root Logger
60     properties.put("log4j.rootLogger", "warn, ikokoon, file");
61     properties.put("log4j.rootCategory", "warn, ikokoon");
62 
63     // Serenity application logging file output
64     properties.put("log4j.appender.file", "org.apache.log4j.DailyRollingFileAppender");
65     properties.put("log4j.appender.file.Threshold", "DEBUG");
66     properties.put("log4j.appender.file.File", SERENITY_LOG_FILE);
67 
68     properties.put("log4j.appender.file.layout", "org.apache.log4j.PatternLayout");
69     properties.put("log4j.appender.file.layout.ConversionPattern", "%d{HH:mm:ss,SSS} %-5p %C:%L - %m%n");
70     properties.put("log4j.appender.file.Append", "false");
71 
72     // Serenity application logging console output
73     properties.put("log4j.appender.ikokoon", "org.apache.log4j.ConsoleAppender");
74     properties.put("log4j.appender.ikokoon.Threshold", "DEBUG");
75     properties.put("log4j.appender.ikokoon.ImmediateFlush", "true");
76     properties.put("log4j.appender.ikokoon.layout", "org.apache.log4j.PatternLayout");
77     properties.put("log4j.appender.ikokoon.layout.ConversionPattern", "%d{HH:mm:ss,SSS} %-5p %C:%L - %m%n");
78 
79     // Set the Serenity categories and thresholds
80     properties.put("log4j.category.net", "WARN");
81     properties.put("log4j.category.com", "WARN");
82     properties.put("log4j.category.org", "WARN");
83 
84     // Specific thresholds
85     properties.put("log4j.category.com.ikokoon", "warn");
86     properties.put("log4j.category.com.ikokoon.serenity", "warn");
87     return properties;
88   }
89 
90 }
