01 package com.ikokoon.serenity.process;
02 
03 import org.apache.log4j.Logger;
04 
05 /**
06  * This class is the process that calls the child processes in the chain. At the end of the coverage processing, i.e. when the unit tests are finished
07  * the processes for collecting the dependency and aggregating the data are started. Each process is chained to the next. The order of the processes
08  * is somewhat important. The accumulator should run first, then the aggregator then the writer.
09  * 
10  * @author Michael Couck
11  * @since 23.08.09
12  * @version 01.00
13  */
14 public abstract class AProcess implements IProcess {
15 
16   /** The logger for the class. */
17   protected Logger logger = Logger.getLogger(this.getClass());
18   /** The child process in the chain if there is one. */
19   private IProcess child;
20 
21   /**
22    * Constructor takes the parent, and the child calls the parent to add it's self to the parent.
23    * 
24    * @param parent
25    *            the parent of this process
26    */
27   public AProcess(IProcess parent) {
28     if (parent != null) {
29       parent.setChild(this);
30     }
31   }
32 
33   /**
34    * This is the method that executes the logic for the process. It will be called by the parent after the parent has executed it's own logic.
35    * 
36    * @param <T>
37    *            the model object, in this case the Document object that collects all the data
38    * @param t
39    *            the document that has all the collected data to date
40    */
41   public void execute() {
42     if (child != null) {
43       child.execute();
44     }
45   }
46 
47   /**
48    * Sets the child of this process. Typically the child calls the parenht to set it's self as the child, in the constructor
49    * 
50    * @param child
51    *            the child to set in the parent
52    */
53   public void setChild(IProcess child) {
54     this.child = child;
55   }
56 
57 }
