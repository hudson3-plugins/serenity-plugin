03 import java.util.ArrayList;
04 import java.util.Arrays;
05 import java.util.List;
06 
07 import org.apache.log4j.Logger;
08 
09 /**
10  * This class generates the permutations for parameters.
11  * 
12  * @author Michael Couck
13  * @since 06.10.09
14  * @version 01.00
15  */
16 public class Permutations {
17 
18   private Logger logger = Logger.getLogger(Permutations.class);
19 
20   public void getPermutations(Object[] objects) {
21     int N = objects.length;
22     Object[] newObjects = new Object[N];
23     for (int i = 0; i < N; i++) {
24       newObjects[i] = objects[i];
25     }
26     List<Object[]> permutations = new ArrayList<Object[]>();
27     getPermutations(newObjects, permutations, N);
28   }
29 
30   public void getPermutations(Object[] objects, List<Object[]> permutations, int n) {
31     if (n == 1) {
32       for (Object object : objects) {
33         logger.debug(object + ":");
34       }
35       permutations.add(Arrays.copyOf(objects, objects.length));
36       return;
37     }
38     for (int i = 0; i < n; i++) {
39       swap(objects, i, n - 1);
40       getPermutations(objects, permutations, n - 1);
41       swap(objects, i, n - 1);
42     }
43   }
44 
45   private Object[] swap(Object[] objects, int i, int j) {
46     Object s = objects[i];
47     objects[i] = objects[j];
48     objects[j] = s;
49     return objects;
50   }
51 
52 }
