006 import java.io.InputStream;

008 import java.util.Enumeration;
009 import java.util.List;

011 import java.util.StringTokenizer;
012 import java.util.TreeSet;
013 import java.util.jar.JarEntry;
014 import java.util.jar.JarFile;
015 import java.util.zip.ZipEntry;

017 import org.objectweb.asm.ClassVisitor;
018 
019 import com.ikokoon.serenity.Configuration;

023 /**
024  * This class looks through the classpath and collects metrics on the classes that were not instanciated by the classloader during the unit tests and
025  * creates a visitor chain for the class that will collect the complexity and dependency metrics for the class.
026  * 
027  * @author Michael Couck
028  * @since 24.07.09
029  * @version 01.00
030  */
031 public class Accumulator extends AProcess {
032 
033   /** The set of jars that are processed so we don't do the same jar more than once. */
034   private Set<String> jarsProcessed = new TreeSet<String>();
035   /** The set of classes that are processed so we don't process the files more than once. */
036   private Set<String> filesProcessed = new TreeSet<String>();
037   /** The chain of adapters for analysing the classes. */
038   private java.lang.Class<ClassVisitor>[] CLASS_ADAPTER_CLASSES;
039 
040   /**
041    * {@inheritDoc}
042    */
043   @SuppressWarnings("unchecked")
044   public Accumulator(IProcess parent) {
045     super(parent);
046     CLASS_ADAPTER_CLASSES = Configuration.getConfiguration().classAdapters
047         .toArray(new java.lang.Class[Configuration.getConfiguration().classAdapters.size()]);
048   }
049 
050   /**
051    * {@inheritDoc}
052    */
053   public void execute() {
054     super.execute();
055     String classpath = Configuration.getConfiguration().getClassPath();
056     StringTokenizer stringTokenizer = new StringTokenizer(classpath, File.pathSeparator);
057     while (stringTokenizer.hasMoreTokens()) {
058       String token = stringTokenizer.nextToken();
059       File file = new File(token);
060       logger.debug("Processing jar : " + token + ", file : " + file.getAbsolutePath());
061       if (!file.exists() || !file.canRead()) {
062         logger.warn("Can't read file : " + file.getAbsolutePath());
063         continue;
064       }
065       if (file.isFile()) {
066         if (token.endsWith(".jar") || token.endsWith(".zip") || token.endsWith(".war") || token.endsWith(".ear")) {
067           logger.debug("Processing jar : " + file.getAbsolutePath());
068           processJar(file);
069         }
070       } else if (file.isDirectory()) {
071         processDir(file);
072       }
073     }
074     // Look for all jars below this directory to find some source
075     List<File> list = new ArrayList<File>();
076     File baseDirectory = new File(".");
077     logger.info("Looking for source in base directory : " + baseDirectory.getAbsolutePath());
078     Toolkit.findFiles(baseDirectory, new Toolkit.IFileFilter() {
079       public boolean matches(File file) {
080         if (file.getName().endsWith(".jar") || file.getName().endsWith(".zip")) {
081           return true;
082         }
083         return false;
084       }
085     }, list);
086     for (File file : list) {
087       logger.debug("Processing jar : " + file.getAbsolutePath());
088       processJar(file);
089     }
090   }
091 
092   /**
093    * Processes a directory on a file system, looks for class files and feeds the byte code into the adapter chain for collecting the metrics for the
094    * class.
095    * 
096    * @param file
097    *            the directory or file to look in for the class data
098    */
099   void processDir(File file) {
100     // Iteratively go through the directories
101     if (file == null || !file.exists() || !file.canWrite()) {
102       return;
103     }
104     if (file.isDirectory()) {
105       File files[] = file.listFiles();
106       for (int j = 0; j < files.length; j++) {
107         file = files[j];
108         processDir(file);
109       }
110     } else if (file.isFile() && file.canRead()) {
111       String filePath = file.getAbsolutePath();
112       filePath = Toolkit.slashToDot(filePath);
113       if (excluded(filePath)) {
114         return;
115       }
116       byte[] classBytes = Toolkit.getContents(file).toByteArray();
117       ByteArrayOutputStream source = new ByteArrayOutputStream();
118 
119       String className = null;
120 
121       // Strip the beginning of the path off the name
122       for (String packageName : Configuration.getConfiguration().includedPackages) {
123         if (filePath.indexOf(packageName) > -1) {
124           int indexOfPackageName = filePath.indexOf(packageName);
125           int classIndex = filePath.lastIndexOf(".class");
126           try {
127             if (classIndex > -1) {
128               className = filePath.substring(indexOfPackageName, classIndex);
129               break;
130             }
131           } catch (Exception e) {
132             logger.error("Exception reading the class files in a directory", e);
133           }
134         }
135       }
136       if (!filesProcessed.add(className)) {
137         return;
138       }
139       processClass(className, classBytes, source);
140     }
141   }
142 
143   /**
144    * Processes a jar or zip file or something like it, looks for class and feeds the byte code into the adapter chain for collecting the metrics for
145    * the class.
146    * 
147    * @param file
148    *            the file to look in for the class data
149    */
150   private void processJar(File file) {
151     // Don't process the jars more than once
152     if (!jarsProcessed.add(file.getName())) {
153       return;
154     }
155 
156     JarFile jarFile = null;
157     try {
158       jarFile = new JarFile(file);
159     } catch (Exception e) {
160       logger.error("Exeption accessing the jar : " + file, e);
161       return;
162     }
163     Enumeration<JarEntry> jarEntries = jarFile.entries();
164     while (jarEntries.hasMoreElements()) {
165       JarEntry jarEntry = jarEntries.nextElement();
166       String entryName = jarEntry.getName();
167       if (excluded(Toolkit.slashToDot(entryName))) {
168         continue;
169       }
170       // logger.debug("Processsing entry : " + entryName);
171       try {
172         InputStream inputStream = jarFile.getInputStream(jarEntry);
173         byte[] classFileBytes = Toolkit.getContents(inputStream).toByteArray();
174         ByteArrayOutputStream source = null;
175         if (jarEntry.getName().indexOf("$") == -1) {
176           source = getSource(jarFile, entryName);
177         } else {
178           source = new ByteArrayOutputStream();
179         }
180         processClass(Toolkit.slashToDot(entryName), classFileBytes, source);
181       } catch (IOException e) {
182         logger.error("Exception reading entry : " + jarEntry + ", from file : " + jarFile, e);
183       }
184     }
185   }
186 
187   private ByteArrayOutputStream getSource(JarFile jarFile, String entryName) throws IOException {
188     // Look for the source
189     String javaEntryName = entryName.substring(0, entryName.lastIndexOf('.')) + ".java";
190     // logger.warn("Looking for source : " + javaEntryName + ", " + entryName);
191     ZipEntry javaEntry = jarFile.getEntry(javaEntryName);
192     if (javaEntry != null) {
193       // logger.warn("Got source : " + javaEntry);
194       InputStream inputStream = jarFile.getInputStream(javaEntry);
195       return Toolkit.getContents(inputStream);
196     }
197     return new ByteArrayOutputStream();
198   }
199 
200   private void processClass(String name, byte[] classBytes, ByteArrayOutputStream source) {
201     if (name != null && name.endsWith(".class")) {
202       name = name.substring(0, name.lastIndexOf('.'));
203     }
204     try {
205       VisitorFactory.getClassVisitor(CLASS_ADAPTER_CLASSES, name, classBytes, source);
206     } catch (Exception e) {
207       int sourceLength = source != null ? source.size() : 0;
208       logger.warn("Class name : " + name + ", length : " + classBytes.length + ", source : " + sourceLength);
209       logger.error("Exception accululating data on class " + name, e);
210     }
211   }
212 
213   private boolean excluded(String name) {
214     // Don't process anything that is not a class file or a Java file
215     if (!name.endsWith(".class")) {
216       logger.debug("Not processing file : " + name);
217       return true;
218     }
219     // Check that the class is included in the included packages
220     if (!Configuration.getConfiguration().included(name)) {
221       logger.debug("File not included : " + name);
222       return true;
223     }
224     // Don't do excluded classes and packages
225     if (Configuration.getConfiguration().excluded(name)) {
226       logger.debug("Excluded file : " + name);
227       return true;
228     }
229     // Don't process the same class twice
230     if (!filesProcessed.add(name)) {
231       logger.debug("Already done file : " + name);
232       return true;
233     }
234     return false;
235   }
