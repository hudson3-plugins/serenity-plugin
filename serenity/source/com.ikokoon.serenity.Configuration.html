020 /**
021  * The configuration object holds the parameters for the processing, some from the system parameters that can be set by the user and some internal
022  * like packages to be excluded always for example java.lang.
023  * 
024  * @author Michael Couck
025  * @since 05.10.09
026  * @version 01.00
027  */
028 public class Configuration {
029 
030   /** The instance of the configuration. */
031   private static Configuration configuration = new Configuration();
032 
033   /** The logger for the class. */
034   public Logger logger;
035   /** Packages that are included in the enhancement. */
036   public Set<String> includedPackages = new TreeSet<String>();
037   /** Patterns in class names that are excluded from enhancement. */
038   public Set<String> excludedPackages = new TreeSet<String>();
039   /** The class adapters that the system will chain. */
040   public List<Class<ClassVisitor>> classAdapters = new ArrayList<Class<ClassVisitor>>();
041 
042   /**
043    * System wide access to the configuration.
044    * 
045    * @return the configuration for the system
046    */
047   public static synchronized Configuration getConfiguration() {
048     return configuration;
049   }
050 
051   /**
052    * Private access insures singularity.
053    */
054   private Configuration() {
055     LoggingConfigurator.configure();
056     logger = Logger.getLogger(this.getClass());
057     addIncludedPackages();
058     addExcludedPackages();
059     addIncludedClassAdapters();
060     addDefaultExcludedPackages();
061   }
062 
063   /**
064    * Checks to see that the class name is included in the packages that are to be included.
065    * 
066    * @param string
067    *            the string to check for pattern inclusion
068    * @return whether the string is included in the pattern list
069    */
070   public boolean included(String string) {
071     if (string == null) {
072       return false;
073     }
074     string = Toolkit.slashToDot(string);
075     for (String pattern : includedPackages) {
076       // logger.info("String : " + string + ", pattern : " + pattern + ", index of : " + (string.indexOf(pattern) > -1));
077       if (string.indexOf(pattern) > -1) {
078         return true;
079       }
080     }
081     return false;
082   }
083 
084   /**
085    * Checks to see if the class is included in the classes that can be enhanced, so for example java.lang is excluded.
086    * 
087    * @param className
088    *            the name of the class to check
089    * @return whether the class is excluded and should not be used
090    */
091   public boolean excluded(String string) {
092     if (string == null) {
093       return true;
094     }
095     string = Toolkit.slashToDot(string);
096     for (String pattern : excludedPackages) {
097       if (string.indexOf(pattern) > -1) {
098         return true;
099       }
100     }
101     return false;
102   }
103 
104   /**
105    * Access to the system properties. This method can be extended to include other properties like in files etc.
106    * 
107    * @param name
108    *            the name of the property
109    * @return the system property with the specified name
110    */
111   public String getProperty(String name) {
112     return System.getProperty(name);
113   }
114 
115   /**
116    * Access to the classpath of the system. Included in the classpath are the jars that were manually added by the user.
117    * 
118    * @return the classpath of the system including the Surefire classpath
119    */
120   public String getClassPath() {
121     StringBuilder builder = new StringBuilder();
122     String classpath = System.getProperty(IConstants.JAVA_CLASS_PATH);
123     builder.append(classpath);
124 
125     String surefireClasspath = System.getProperty(IConstants.SUREFIRE_TEST_CLASS_PATH);
126     if (surefireClasspath != null) {
127       builder.append(File.pathSeparator);
128       builder.append(surefireClasspath);
129     }
130 
131     String includedJars = System.getProperty(IConstants.INCLUDED_JARS_PROPERTY);
132     logger.debug("Included jars : " + includedJars);
133     if (includedJars != null) {
134       StringTokenizer tokenizer = new StringTokenizer(includedJars, ";:", false);
135       while (tokenizer.hasMoreTokens()) {
136         String jarFile = tokenizer.nextToken();
137         builder.append(File.pathSeparator);
138         builder.append(jarFile);
139       }
140     }
141     return builder.toString();
142   }
143 
144   private void addIncludedPackages() {
145     String packageNames = System.getProperty(IConstants.INCLUDED_PACKAGES_PROPERTY);
146     logger.debug("Package names : " + packageNames);
147     if (packageNames != null) {
148       StringTokenizer tokenizer = new StringTokenizer(packageNames, ";: ", false);
149       while (tokenizer.hasMoreTokens()) {
150         String packageName = tokenizer.nextToken();
151         packageName = Toolkit.stripWhitespace(packageName);
152         includedPackages.add(packageName);
153         logger.debug("Added package to enhance : " + packageName);
154       }
155     }
156   }
157 
158   private void addExcludedPackages() {
159     String excludedPatterns = System.getProperty(IConstants.EXCLUDED_PACKAGES_PROPERTY);
160     if (excludedPatterns != null) {
161       StringTokenizer tokenizer = new StringTokenizer(excludedPatterns, ";: ", false);
162       while (tokenizer.hasMoreTokens()) {
163         String excludedPattern = tokenizer.nextToken();
164         excludedPackages.add(excludedPattern);
165       }
166     }
167   }
168 
169   @SuppressWarnings("unchecked")
170   private void addIncludedClassAdapters() {
171     String adapterNames = System.getProperty(IConstants.INCLUDED_ADAPTERS_PROPERTY);
172     if (adapterNames != null) {
173       StringTokenizer tokenizer = new StringTokenizer(adapterNames, ";: ", false);
174       while (tokenizer.hasMoreTokens()) {
175         String adapterName = tokenizer.nextToken();
176         adapterName = Toolkit.stripWhitespace(adapterName);
177         try {
178           if (adapterName.equals(IConstants.COVERAGE)) {
179             classAdapters.add((Class<ClassVisitor>) Class.forName(CoverageClassAdapter.class.getName()));
180           }
181           if (adapterName.equals(IConstants.COMPLEXITY)) {
182             classAdapters.add((Class<ClassVisitor>) Class.forName(ComplexityClassAdapter.class.getName()));
183           }
184           if (adapterName.equals(IConstants.DEPENDENCY)) {
185             classAdapters.add((Class<ClassVisitor>) Class.forName(DependencyClassAdapter.class.getName()));
186           }
187           if (adapterName.equals(IConstants.PROFILING)) {
188             classAdapters.add((Class<ClassVisitor>) Class.forName(ProfilingClassAdapter.class.getName()));
189           }
190         } catch (ClassNotFoundException e) {
191           logger.error("Class : " + adapterName + " not found", e);
192         }
193       }
194     }
195   }
196 
197   private void addDefaultExcludedPackages() {
198     excludedPackages.add("java.lang");
199     excludedPackages.add("sun");
200     excludedPackages.add("sunw");
201     excludedPackages.add("com.sun");
202     excludedPackages.add("Test");
203     excludedPackages.add(Project.class.getPackage().getName());
204   }
