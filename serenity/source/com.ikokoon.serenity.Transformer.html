027 /**
028  * This class is the entry point for the Serenity code coverage/complexity/dependency/profiling functionality. This class is called by the JVM on
029  * startup. The agent then has first access to the byte code for all classes that are loaded. During this loading the byte code can be enhanced.
030  * 
031  * @author Michael Couck
032  * @since 12.07.09
033  * @version 01.00
034  */
035 public class Transformer implements ClassFileTransformer, IConstants {
036 
037   /** The logger. */
038   private static Logger LOGGER;
039   /** During tests there can be more than one shutdown hook added. */
040   private static boolean INITIALISED = false;
041   /** The chain of adapters for analysing the classes. */
042   private static Class<ClassVisitor>[] CLASS_ADAPTER_CLASSES;
043 
044   /**
045    * This method is called by the JVM at startup. This method will only be called if the command line for starting the JVM has the following on it:
046    * -javaagent:serenity/serenity.jar. This instruction tells the JVM that there is an agent that must be used. In the META-INF directory of the jar
047    * specified there must be a MANIFEST.MF file. In this file the instructions must be something like the following:
048    * 
049    * Manifest-Version: 1.0 <br>
050    * Boot-Class-Path: asm-3.1.jar and so on..., in the case that the required libraries are not on the classpath, which they should be<br>
051    * Premain-Class: com.ikokoon.serenity.Transformer
052    * 
053    * Another line in the manifest can start an agent after the JVM has been started, but not for all JVMs. So not very useful.
054    * 
055    * These instructions tell the JVM to call this method when loading class files.
056    * 
057    * @param args
058    *            a set of arguments that the JVM will call the method with
059    * @param instrumentation
060    *            the instrumentation implementation of the JVM
061    */
062   @SuppressWarnings("unchecked")
063   public static void premain(String args, Instrumentation instrumentation) {
064     if (!INITIALISED) {
065       INITIALISED = true;
066 
067       // printSystemProperties();
068 
069       LoggingConfigurator.configure();
070       CLASS_ADAPTER_CLASSES = Configuration.getConfiguration().classAdapters.toArray(new Class[Configuration.getConfiguration().classAdapters
071           .size()]);
072       LOGGER = Logger.getLogger(Transformer.class);
073 
074       Runtime.getRuntime().addShutdownHook(new Thread() {
075         public void run() {
076           Date start = new Date();
077           LOGGER.info("Starting accumulation : " + start);
078           IDataBase dataBase = IDataBase.DataBaseManager.getDataBase(DataBaseRam.class, IConstants.DATABASE_FILE_RAM, false, null);
079 
080           long processStart = System.currentTimeMillis();
081           new Accumulator(null).execute();
082           LOGGER.info("Accululator : " + (System.currentTimeMillis() - processStart));
083 
084           processStart = System.currentTimeMillis();
085           new Cleaner(null, dataBase).execute();
086           LOGGER.info("Cleaner : " + (System.currentTimeMillis() - processStart));
087 
088           processStart = System.currentTimeMillis();
089           new Aggregator(null, dataBase).execute();
090           LOGGER.info("Aggregator : " + (System.currentTimeMillis() - processStart));
091 
092           // DataBaseToolkit.dump(dataBase, null, "Transformer database dump before pruning");
093 
094           processStart = System.currentTimeMillis();
095           new Pruner(null, dataBase).execute();
096           LOGGER.info("Pruner : " + (System.currentTimeMillis() - processStart));
097 
098           processStart = System.currentTimeMillis();
099           if (!dataBase.isClosed()) {
100             dataBase.close();
101           }
102           LOGGER.info("Close database : " + (System.currentTimeMillis() - processStart));
103 
104           Date end = new Date();
105           long million = 1000 * 1000;
106           long duration = end.getTime() - start.getTime();
107           LOGGER.info("Finished accumulation : " + end + ", duration : " + duration + " millis");
108           LOGGER.info("Total memory : " + (Runtime.getRuntime().totalMemory() / million) + ", max memory : "
109               + (Runtime.getRuntime().maxMemory() / million) + ", free memory : " + (Runtime.getRuntime().freeMemory() / million));
110         }
111       });
112 
113       if (instrumentation != null) {
114         instrumentation.addTransformer(new Transformer());
115       }
116       String cleanClasses = Configuration.getConfiguration().getProperty(IConstants.CLEAN_CLASSES);
117       if (cleanClasses != null && cleanClasses.equals(Boolean.TRUE.toString())) {
118         File serenityDirectory = new File(IConstants.SERENITY_DIRECTORY);
119         Toolkit.deleteFiles(serenityDirectory, ".class");
120         if (!serenityDirectory.exists()) {
121           if (!serenityDirectory.mkdirs()) {
122             LOGGER.warn("Didn't re-create Serenity directory : " + serenityDirectory.getAbsolutePath());
123           }
124         }
125       }
126       // This is the underlying database that will persist the data to the file system
127       IDataBase odbDataBase = IDataBase.DataBaseManager.getDataBase(DataBaseOdb.class, IConstants.DATABASE_FILE_ODB, true, null);
128       // And we try the JPA database for the activation depth
129       // IDataBase jpaDataBase = IDataBase.DataBaseManager.getDataBase(DataBaseJpa.class, IConstants.DATABASE_FILE_JPA, true, null);
130       // This is the ram database that will hold all the data in memory for better performance
131       IDataBase ramDataBase = IDataBase.DataBaseManager.getDataBase(DataBaseRam.class, IConstants.DATABASE_FILE_RAM, true, odbDataBase);
132       // Just in case the database is opened on a file that is not empty
133       DataBaseToolkit.clear(ramDataBase);
134     }
135   }

137   protected static void printSystemProperties() {
138     System.out.println("Working directory : " + new File(".").getAbsolutePath());
139     System.out.println("Java class path : " + System.getProperty("java.class.path"));
140     Enumeration<Object> keys = System.getProperties().keys();
141     while (keys.hasMoreElements()) {
142       Object key = keys.nextElement();
143       if (key == null) {
144         continue;
145       }
146       System.out.println("System property : " + key + "=" + System.getProperties().getProperty(key.toString()));
147     }
148   }

150   /**
151    * This method transforms the classes that are specified.
152    */
153   public byte[] transform(ClassLoader loader, String className, Class<?> classBeingRedefined, ProtectionDomain protectionDomain, byte[] classBytes)
154       throws IllegalClassFormatException {
155     // Can we implement a classloader here? Would it make things simpler/more robust/faster?
156     // Thread.currentThread().setContextClassLoader(and the custom classloader);
157     if (loader != ClassLoader.getSystemClassLoader()) {
158       LOGGER.debug("No system classloader : " + className);
159       return classBytes;
160     }
161     if (Configuration.getConfiguration().excluded(className)) {
162       LOGGER.debug("Excluded class : " + className);
163       return classBytes;
164     }
165     if (Configuration.getConfiguration().included(className)) {
166       LOGGER.debug("Enhancing class : " + className);
167       ByteArrayOutputStream source = new ByteArrayOutputStream(0);
168       ClassWriter writer = (ClassWriter) VisitorFactory.getClassVisitor(CLASS_ADAPTER_CLASSES, className, classBytes, source);
169       byte[] enhancedClassBytes = writer.toByteArray();
170       String writeClasses = Configuration.getConfiguration().getProperty(IConstants.WRITE_CLASSES);
171       if (writeClasses != null && writeClasses.equals(Boolean.TRUE.toString())) {
172         writeClass(className, enhancedClassBytes);
173       }
174       return enhancedClassBytes;
175     } else {
176       LOGGER.debug("Class not included : " + className);
177     }
178     return classBytes;
179   }
180 
181   private void writeClass(String className, byte[] classBytes) {
182     // Write the class so we can check it with JD decompiler visually
183     String directoryPath = Toolkit.dotToSlash(Toolkit.classNameToPackageName(className));
184     String fileName = className.replaceFirst(Toolkit.classNameToPackageName(className), "") + ".class";
185     File directory = new File(IConstants.SERENITY_DIRECTORY + File.separator + directoryPath);
186     if (!directory.exists()) {
187       directory.mkdirs();
188     }
189     LOGGER.debug(directory.getAbsolutePath());
190 
191     File file = new File(directory, fileName);
192     Toolkit.setContents(file, classBytes);
193   }
