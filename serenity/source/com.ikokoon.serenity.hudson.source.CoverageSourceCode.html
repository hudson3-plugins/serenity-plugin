02 
03 import java.io.ByteArrayInputStream;
04 import java.io.InputStream;
05 import java.io.InputStreamReader;
06 import java.io.StringWriter;
07 
08 import org.apache.log4j.Logger;
09 
10 import com.ikokoon.serenity.model.Class;
11 
12 import de.java2html.converter.JavaSource2HTMLConverter;
13 import de.java2html.javasource.JavaSource;
14 import de.java2html.javasource.JavaSourceParser;
15 import de.java2html.options.JavaSourceConversionOptions;
16 
17 /**
18  * This class takes the source for a Java file and generates HTML from the source that can be displayed in a browser. This class uses the Java2Html
19  * package from java2html.de.
20  * 
21  * @author Michael Couck
22  * @since 09.12.09
23  * @version 01.00
24  */
25 public class CoverageSourceCode implements ISourceCode {
26 
27   private Logger logger = Logger.getLogger(CoverageSourceCode.class);
28   /** The class from the model that has the source code in it. */
29   private Class<?, ?> klass;
30   private String source;
31   private JavaSourceParser javaSourceParser;
32 
33   /**
34    * This class is for instantiating the options. In the transformer(the agent) this throws an exception for some reason so this is a work around,
35    * or otherwise known as a hack.
36    * 
37    * @author Michael Couck
38    * 
39    */
40   class JavaSourceParserExt extends JavaSourceParser {
41     public JavaSourceParserExt() {
42       super(JavaSourceConversionOptions.getRawDefault());
43     }
44   }
45 
46   private JavaSourceConversionOptions options = JavaSourceConversionOptions.getRawDefault();
47   {
48     options.setAddLineAnchors(true);
49     options.setShowLineNumbers(true);
50   }
51 
52   /**
53    * Constructor takes the class that contains the source for the conversion.
54    * 
55    * @param klass
56    *            the class that contains the source
57    * @param the
58    *            source of the Java class
59    */
60   public CoverageSourceCode(Class<?, ?> klass, String source) {
61     this.klass = klass;
62     this.source = source;
63     try {
64       javaSourceParser = new JavaSourceParserExt();
65     } catch (Exception e) {
66       logger.info("Exception initialising the Java source parser", e);
67     }
68   }
69 
70   /**
71    * {@inheritDoc}
72    */
73   public String getSource() {
74     if (this.klass != null && this.source != null) {
75       try {
76         // Convert the Java source to HTML
77         InputStream inputStream = new ByteArrayInputStream(source.getBytes());
78         JavaSource javaSource = javaSourceParser.parse(new InputStreamReader(inputStream));
79         JavaSource2HTMLConverter converter = new JavaSource2HTMLConverterExt(klass);
80         StringWriter writer = new StringWriter();
81         converter.convert(javaSource, options, writer);
82         String html = writer.toString();
83         return html;
84       } catch (Exception e) {
85         logger.error("Exception generating the HTML for the class source : " + klass, e);
86       }
87     }
88     return "No source";
89   }
