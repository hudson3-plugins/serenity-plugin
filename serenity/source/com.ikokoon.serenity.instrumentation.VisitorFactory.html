001 package com.ikokoon.serenity.instrumentation;
002 
003 import java.io.ByteArrayOutputStream;
004 
005 import org.apache.log4j.Logger;
006 import org.objectweb.asm.AnnotationVisitor;
007 import org.objectweb.asm.ClassReader;
008 import org.objectweb.asm.ClassVisitor;
009 import org.objectweb.asm.ClassWriter;
010 import org.objectweb.asm.FieldVisitor;
011 import org.objectweb.asm.MethodVisitor;
012 import org.objectweb.asm.signature.SignatureReader;
013 import org.objectweb.asm.signature.SignatureVisitor;
014 
015 import com.ikokoon.serenity.instrumentation.dependency.DependencyAnnotationAdapter;
016 import com.ikokoon.serenity.instrumentation.dependency.DependencySignatureAdapter;
017 import com.ikokoon.toolkit.ObjectFactory;
018 
019 /**
020  * This class instantiates visitors for classes, methods, field, signature and annotations.
021  * 
022  * @author Michael Couck
023  * @since 09.12.09
024  * @version 01.00
025  */
026 public class VisitorFactory {
027 
028   private static Logger LOGGER = Logger.getLogger(VisitorFactory.class);
029 
030   /**
031    * Instantiates a chain of class visitors. Each visitor can modify or add code to the class as it is parsed and the writer will output the new
032    * class byte code.
033    * 
034    * @param classAdapterClasses
035    *            the class visitor classes
036    * @param className
037    *            the name of the class to be visited
038    * @param classBytes
039    *            the byte array of the byte code
040    * @param source
041    *            the output stream of the source code for the class
042    * @return the class visitor/writer
043    */
044   public static ClassVisitor getClassVisitor(Class<ClassVisitor>[] classAdapterClasses, String className, byte[] classBytes,
045       ByteArrayOutputStream source) {
046     ClassReader reader = new ClassReader(classBytes);
047     ClassWriter writer = new ClassWriter(reader, ClassWriter.COMPUTE_MAXS);
048     ClassVisitor visitor = writer;
049     for (Class<ClassVisitor> klass : classAdapterClasses) {
050       Object[] parameters = new Object[] { visitor, className, classBytes, source };
051       visitor = ObjectFactory.getObject(klass, parameters);
052       LOGGER.debug("Adding class visitor : " + visitor);
053     }
054     reader.accept(visitor, 0);
055     return writer;
056   }
057 
058   /**
059    * This method constructs a method visitor chain based on the class of the method adapter passed as a parameter.
060    * 
061    * @param visitor
062    *            the parent method visitor
063    * @param klass
064    *            the class of the method visitor to initialise
065    * @param className
066    *            the name of the class that the method visitor will visit
067    * @param name
068    *            the name of the method
069    * @param desc
070    *            the description or signature of the method in byte code format
071    * @return the method visitor
072    */
073   public static MethodVisitor getMethodVisitor(MethodVisitor visitor, Class<?> klass, String className, String name, String desc) {
074     Object[] parameters = new Object[] { visitor, className, name, desc };
075     MethodVisitor adapter = (MethodVisitor) ObjectFactory.getObject(klass, parameters);
076     return adapter;
077   }
078 
079   /**
080    * This method constructs a field visitor chain that will visit the byte code for a field in a class.
081    * 
082    * @param visitor
083    *            the parent field visitor
084    * @param klass
085    *            the field visitor class type
086    * @param className
087    *            the name of the class the field is in
088    * @param description
089    *            the description of the field in byte code format
090    * @param signature
091    *            the signature of the field in byte code
092    * @return the field visitor
093    */
094   public static FieldVisitor getFieldVisitor(FieldVisitor visitor, Class<?> klass, String className, String description, String signature) {
095     Object[] parameters = new Object[] { visitor, className, description, signature };
096     FieldVisitor adapter = (FieldVisitor) ObjectFactory.getObject(klass, parameters);
097     return adapter;
098   }
099 
100   /**
101    * This method constructs a signature visitor that will visit a signature for something, could be an annotation or a method.
102    * 
103    * @param className
104    *            the name of the class the signature is in
105    * @param signature
106    *            the signature in byte code to visit
107    * @return the signature to be visited
108    */
109   public static SignatureVisitor getSignatureVisitor(String className, String signature) {
110     SignatureReader reader = new SignatureReader(signature);
111     SignatureVisitor adapter = new DependencySignatureAdapter(className);
112     reader.accept(adapter);
113     return adapter;
114   }
115 
116   /**
117    * This method constructs an annotation visitor to visit annotations.
118    * 
119    * @param visitor
120    *            the parent annotation visitor
121    * @param className
122    *            the name of the class with the annotation
123    * @param description
124    *            the description or signature of the annotation
125    * @return the annotation visitor
126    */
127   public static AnnotationVisitor getAnnotationVisitor(AnnotationVisitor visitor, String className, String description) {
128     AnnotationVisitor adapter = new DependencyAnnotationAdapter(visitor, className, description);
129     return adapter;
130   }
