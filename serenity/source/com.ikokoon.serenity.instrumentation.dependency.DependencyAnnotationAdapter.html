01 package com.ikokoon.serenity.instrumentation.dependency;
02 
03 import org.apache.log4j.Logger;
04 import org.objectweb.asm.AnnotationVisitor;
05 
06 import com.ikokoon.serenity.instrumentation.VisitorFactory;
07 import com.ikokoon.toolkit.Toolkit;
08 
09 /**
10  * This class visits the annotation description collecting the dependency information on the class that defines the annotation.
11  * 
12  * @author Michael Couck
13  * @since 21.11.09
14  * @version 01.00
15  */
16 public class DependencyAnnotationAdapter implements AnnotationVisitor {
17 
18   private Logger logger = Logger.getLogger(this.getClass());
19   /** The parent annotation visitor. */
20   private AnnotationVisitor annotationVisitor;
21   /** The class that has the annotation. */
22   private String className;
23 
24   /**
25    * Constructor initialises a {@link DependencyAnnotationAdapter}, takes the annotation visitor parent, the class name that uses the annotation and
26    * the description of the annotation.
27    * 
28    * @param annotationVisitor
29    *            the parent annotation visitor
30    * @param className
31    *            the class name that uses the annotation
32    * @param description
33    *            the description of the annotation
34    */
35   public DependencyAnnotationAdapter(AnnotationVisitor annotationVisitor, String className, String description) {
36     logger.debug("Class name : " + className + ", " + description);
37     this.className = Toolkit.slashToDot(className);
38     this.annotationVisitor = annotationVisitor;
39     VisitorFactory.getSignatureVisitor(className, description);
40   }
41 
42   /**
43    * {@inheritDoc}
44    */
45   public void visit(String name, Object value) {
46     logger.debug("visit : " + className + ", " + name + ", " + value);
47     if (name != null && value != null) {
48       try {
49         VisitorFactory.getSignatureVisitor(className, value.toString());
50       } catch (StringIndexOutOfBoundsException e) {
51         // We swallow this exception more or less because some annotations may contain
52         // Strings that are not even remotely like a class name, like Remote/bean for example
53         logger.debug("String out of bounds for : " + className + ", " + name + ", " + value);
54       }
55     }
56     if (name != null) {
57       annotationVisitor.visit(name, value);
58     }
59   }
60 
61   /**
62    * {@inheritDoc}
63    */
64   public AnnotationVisitor visitAnnotation(String name, String desc) {
65     logger.debug("visitAnnotation : " + className + ", " + name + ", " + desc);
66     AnnotationVisitor visitor = annotationVisitor.visitAnnotation(name, desc);
67     AnnotationVisitor adapter = VisitorFactory.getAnnotationVisitor(visitor, name, desc);
68     return adapter;
69   }
70 
71   /**
72    * {@inheritDoc}
73    */
74   public AnnotationVisitor visitArray(String name) {
75     logger.debug("visitArray : " + className + ", " + name);
76     return annotationVisitor.visitArray(name);
77   }
78 
79   /**
80    * {@inheritDoc}
81    */
82   public void visitEnd() {
83     logger.debug("visitEnd : " + className);
84     annotationVisitor.visitEnd();
85   }
86 
87   /**
88    * {@inheritDoc}
89    */
90   public void visitEnum(String name, String desc, String value) {
91     logger.debug("visitEnum : " + className + ", " + name + ", " + desc + ", " + value);
92     VisitorFactory.getSignatureVisitor(className, desc);
93     if (name != null) {
94       annotationVisitor.visitEnum(name, desc, value);
95     }
96   }
97 
98 }
